23  double get_beta2(double beta_x, double beta_y, double beta_z)
24  {
25  double beta2 = std::pow(beta_x, 2) + std::pow(beta_y, 2)
26  + std::pow(beta_z, 2);
27  if (beta2 == 1) throw marley::Error(std::string("Cannot perform")
28  + " Lorentz boost because \u03B2^2 = 1 and therefore the Lorentz factor"
29  + "\u03B3 is infinite.");
30  else if (beta2 > 1) throw marley::Error(std::string("Cannot perform")
31  + " Lorentz boost because \u03B2^2 = " + std::to_string(beta2) + " > 1,"
32  + " which is unphysical.");
33  return beta2;
34  }

38 // Rotates a particle's 3-momentum so that it points in the (x, y, z) direction
39 void marley_kinematics::rotate_momentum_vector(double x, double y, double z,
40  marley::Particle& particle_to_rotate)
41 {
42  // Get the magnitude of the vector pointing in the desired direction
43  double r = std::sqrt( std::pow(x, 2) + std::pow(y, 2) + std::pow(z, 2) );
44 
45  // Get magnitude of the particle's momentum
46  double rp = particle_to_rotate.momentum_magnitude();
47 
48  // Rotate the particle's momentum vector into the desired direction
49  double ratio = rp / r;
50 
51  double new_px = x * ratio;
52  double new_py = y * ratio;
53  double new_pz = z * ratio;
54 
55  particle_to_rotate.set_px( new_px );
56  particle_to_rotate.set_py( new_py );
57  particle_to_rotate.set_pz( new_pz );
58 }

60 // Lorentz boost a particle, replacing its energy and momentum with the boosted
61 // versions
62 void marley_kinematics::lorentz_boost(double beta_x, double beta_y,
63  double beta_z, marley::Particle& particle_to_boost)
64 {
65  double beta2 = get_beta2(beta_x, beta_y, beta_z);
66 
67  // If beta is zero in all directions, then we don't need to do the boost at all
68  if ( beta2 == 0. ) return;
69 
70  // Calculate the Lorentz factor based on the boost velocity
71  double gamma = 1. / std::sqrt( 1. - beta2 );
72 
73  double E = particle_to_boost.total_energy();
74  double px = particle_to_boost.px();
75  double py = particle_to_boost.py();
76  double pz = particle_to_boost.pz();
77  double m = particle_to_boost.mass();
78 
79  // Compute the boosted energy and 3-momentum for the particle (the expressions
80  // we use here are based on
81  // https://en.wikipedia.org/wiki/Lorentz_transformation#Boost_in_any_direction)
82  double beta_dot_p = beta_x * px + beta_y * py + beta_z * pz;
83  double factor = ( gamma - 1. ) * beta_dot_p / beta2;
84 
85  double new_E = gamma * ( E - beta_dot_p );
86 
87  // The new energy could conceivably dip slightly below the mass
88  // of the particle due to roundoff errors. If this is the case,
89  // set it to the particle mass.
90  if (new_E < m) new_E = m;
91 
92  double new_px = ( -gamma * E + factor ) * beta_x + px;
93  double new_py = ( -gamma * E + factor ) * beta_y + py;
94  double new_pz = ( -gamma * E + factor ) * beta_z + pz;
95 
96  particle_to_boost.set_total_energy( new_E );
97  particle_to_boost.set_px( new_px );
98  particle_to_boost.set_py( new_py );
99  particle_to_boost.set_pz( new_pz );
100 }

102 // Right now, this function assumes that the coordinate axes in the lab
103 // and initial particle rest frames are coincident (i.e., the coordinate axes
104 // aren't rotated between the two frames at all). Since all interactions in
105 // the current version of the code are isotropic (except for the initial
106 // 2-body neutrino scattering reaction, which occurs for an incident neutrino
107 // traveling in the positive z direction in the lab frame [target nucleus's
108 // rest frame]), this is all we need for now.
109 // TODO: Expand this to allow the two frames to be rotated with respect to each other.
110 void marley_kinematics::two_body_decay(const marley::Particle& initial_particle,
111  marley::Particle& first_product, marley::Particle& second_product,
112  double cos_theta_first, double phi_first)
113 {
114  // Get the masses of all three particles
115  double M = initial_particle.mass();
116  double mfirst = first_product.mass();
117  double msecond = second_product.mass();
118 
119  // Check to make sure the decay is kinematically allowed
120  if (M < mfirst + msecond) throw marley::Error(std::string("A two-body")
121  + " decay was requested that is not kinematically allowed.");
122 
123  double M2 = std::pow( M, 2 );
124  double mfirst2 = std::pow( mfirst, 2 );
125  double msecond2 = std::pow( msecond, 2 );
126 
127  // Compute the energies for the decay products in the rest
128  // frame of the initial particle
129  double Efirst = ( M2 - msecond2 + mfirst2 ) / ( 2 * M );
130  // M - Efirst == (M2 - mfirst2 + msecond2) / (2 * M)
131  double Esecond = M - Efirst;
132 
133  // Avoid roundoff issues by not allowing the energies to dip below
134  // the particle masses
135  if ( Efirst < mfirst ) Efirst = mfirst;
136  if ( Esecond < msecond ) Esecond = msecond;
137 
138  // Compute the 3-momenta for the decay products, still in the initial
139  // particle's rest frame
140  double pfirst = marley_utils::real_sqrt( std::pow(Efirst, 2) - mfirst2 );
141  double sin_theta_first = marley_utils::real_sqrt( 1.
142  - std::pow(cos_theta_first, 2) );
143  double p1x = pfirst * sin_theta_first * std::cos( phi_first );
144  double p1y = pfirst * sin_theta_first * std::sin( phi_first );
145  double p1z = pfirst * cos_theta_first;
146 
147  // Conservation of 3-momenta tells us that, in the rest frame of the
148  // initial particle, p1 = -p2. We can use this as a shortcut.
149  double p2x = -p1x;
150  double p2y = -p1y;
151  double p2z = -p1z;
152 
153  // Now that we have this information, load it into the product
154  // particles.
155  first_product.set_total_energy( Efirst );
156  first_product.set_px( p1x );
157  first_product.set_py( p1y );
158  first_product.set_pz( p1z );
159 
160  second_product.set_total_energy( Esecond );
161  second_product.set_px( p2x );
162  second_product.set_py( p2y );
163  second_product.set_pz( p2z );
164 
165  // Compute the parameters needed to boost these particles
166  // from the initial particle's rest frame into the lab frame
167  double E_i = initial_particle.total_energy();
168  double px_i = initial_particle.px();
169  double py_i = initial_particle.py();
170  double pz_i = initial_particle.pz();
171 
172  // Boost in the opposite direction (this gives us the minus signs below) from
173  // the 3-momentum of rest_frame_particle. This takes the null 3-vector to the
174  // initial particle's 3-momentum.
175  double beta_x = -px_i / E_i;
176  double beta_y = -py_i / E_i;
177  double beta_z = -pz_i / E_i;
178 
179  // Boost both products to the lab frame by replacing their
180  // energies and momenta with the boosted versions
181  lorentz_boost( beta_x, beta_y, beta_z, first_product );
182  lorentz_boost( beta_x, beta_y, beta_z, second_product );
183 }

185 // Get the square of the total energy of two particles in their center of
186 // momentum frame
187 double marley_kinematics::get_mandelstam_s(const marley::Particle& p1,
188  const marley::Particle& p2)
189 {
190  double E1 = p1.total_energy();
191  double m1 = p1.mass();
192  double E2 = p2.total_energy();
193  double m2 = p2.mass();
194 
195  // If one of the particles is at rest, use a shortcut. Otherwise,
196  // Lorentz transform to the center of momentum frame to determine
197  // the total cm frame energy
198  if (E1 == m1) return std::pow(m1, 2) + std::pow(m2, 2) + 2 * m1 * E2;
199  else if (E2 == m2) return std::pow(m1, 2) + std::pow(m2, 2) + 2 * m2 * E1;
200  else {
201  // Get total energy, momentum, and mass values for the two particles
202  double E_tot = E1 + E2;
203  double px_tot = p1.px() + p2.px();
204  double py_tot = p1.py() + p2.py();
205  double pz_tot = p1.pz() + p2.pz();
206  double m_tot = m1 + m2;
207 
208  // Get boost parameters for a Lorentz transform to the center of momentum
209  // frame
210  double beta_x = px_tot / E_tot;
211  double beta_y = py_tot / E_tot;
212  double beta_z = pz_tot / E_tot;
213 
214  // Calculate the Lorentz factor based on the boost velocity
215  double beta2 = get_beta2(beta_x, beta_y, beta_z);
216  double gamma = 1 / std::sqrt(1 - beta2);
217 
218  // Compute the total boosted energy in the cm frame
219  double beta_dot_p_tot = beta_x * px_tot + beta_y * py_tot + beta_z * pz_tot;
220  double E_tot_cm = gamma * (E_tot - beta_dot_p_tot);
221 
222  // The new energy could conceivably dip slightly below the total rest mass
223  // of the particles due to roundoff errors. If this is the case,
224  // set it to the total rest mass.
225  if (E_tot_cm < m_tot) E_tot_cm = m_tot;
226 
227  return std::pow(E_tot_cm, 2);
228  }

233 // Boost two particles into their mutual CM frame
234 void marley_kinematics::boost_to_cm_frame(marley::Particle& p1,
235  marley::Particle& p2)
236 {
237  // Get parameters for a Lorentz boost to the center of momentum frame
238  double E_tot = p1.total_energy() + p2.total_energy();
239  double px_tot = p1.px() + p2.px();
240  double py_tot = p1.py() + p2.py();
241  double pz_tot = p1.pz() + p2.pz();
242 
243  double beta_x = px_tot / E_tot;
244  double beta_y = py_tot / E_tot;
245  double beta_z = pz_tot / E_tot;
246 
247  lorentz_boost(beta_x, beta_y, beta_z, p1);
248  lorentz_boost(beta_x, beta_y, beta_z, p2);
249 }
