29  // Loads the 3-vector dest with the cross product v1 x v2
30  void cross_product(ThreeVector& dest,
31  const ThreeVector& v1, const ThreeVector& v2)
32  {
33  dest[0] = v1[1] * v2[2] - v1[2] * v2[1];
34  dest[1] = v1[2] * v2[0] - v1[0] * v2[2];
35  dest[2] = v1[0] * v2[1] - v1[1] * v2[0];
36  }
37 
38  // Returns the dot product v1 . v2
39  double dot_product(const ThreeVector& v1,
40  const ThreeVector& v2)
41  {
42  return v1[0] * v2[0] + v1[1] * v2[1] + v1[2] * v2[2];
43  }

60 // Returns a copy of the 3-vector v normalized to have unit magnitude
61 ThreeVector marley::RotationMatrix::normalize(const ThreeVector& v)
62 {
63  static ThreeVector nv({0., 0., 0.});
64  double norm_factor = std::sqrt(std::pow(v[0], 2) + std::pow(v[1], 2)
65  + std::pow(v[2], 2));
66  if (norm_factor <= 0.) throw marley::Error(std::string("Invalid vector")
67  + " magnitude encountered in marley::RotationMatrix::normalize()");
68  else norm_factor = 1. / norm_factor;
69  nv[0] = norm_factor * v[0];
70  nv[1] = norm_factor * v[1];
71  nv[2] = norm_factor * v[2];
72  return nv;
73 }

111 marley::RotationMatrix::RotationMatrix(const ThreeVector& from_vec,
112  const ThreeVector& to_vec)
113 {
114  static constexpr ThreeVector null_three_vector = { 0., 0., 0. };
115 
116  if (from_vec == null_three_vector)
117  throw marley::Error(std::string("Null from vector")
118  + " passed to constructor of marley::RotationMatrix");
119  else if (to_vec == null_three_vector)
120  throw marley::Error(std::string("Null to vector")
121  + " passed to constructor of marley::RotationMatrix");
122 
123  // The from and to vectors must be normalized to use this method, so make any
124  // necessary adjustments now.
125  ThreeVector from = normalize(from_vec);
126  ThreeVector to = normalize(to_vec);
127 
128  double e = dot_product(from, to);
129  double f = std::abs(e);
130 
131  static constexpr double EPSILON = 0.000001;
132  if (f > 1.0 - EPSILON) { // "from" and "to" vectors are almost parallel
133 
134  // Temporary storage vectors
135  ThreeVector v;
136  ThreeVector u;
137 
138  // Find the standard unit vector x most nearly orthogonal to "from"
139  ThreeVector x;
140  x[0] = std::abs(from[0]);
141  x[1] = std::abs(from[1]);
142  x[2] = std::abs(from[2]);
143 
144  if (x[0] < x[1])
145  {
146  if (x[0] < x[2])
147  {
148  x[0] = 1.0; x[1] = x[2] = 0.0;
149  }
150  else
151  {
152  x[2] = 1.0; x[0] = x[1] = 0.0;
153  }
154  }
155  else
156  {
157  if (x[1] < x[2])
158  {
159  x[1] = 1.0; x[0] = x[2] = 0.0;
160  }
161  else
162  {
163  x[2] = 1.0; x[0] = x[1] = 0.0;
164  }
165  }
166 
167  // u = x - from
168  subtract(u, x, from);
169 
170  // v = x - to
171  subtract(v, x, to);
172 
173  // coefficients for later use
174  double c1 = 2.0 / dot_product(u, u);
175  double c2 = 2.0 / dot_product(v, v);
176  double c3 = c1 * c2 * dot_product(u, v);
177 
178  for (unsigned i = 0; i < 3; ++i) {
179  for (unsigned j = 0; j < 3; ++j) {
180  matrix_[i][j] = - c1 * u[i] * u[j] - c2 * v[i] * v[j]
181  + c3 * v[i] * u[j];
182  }
183  matrix_[i][i] += 1.0;
184  }
185  }
186  else // the most common case, unless "from" = "to", or "from" = -"to"
187  {
188  ThreeVector v;
189  cross_product(v, from, to); // v = from x to
190 
191  // hand-optimized version (9 mults less than original)
192  // optimization by Gottfried Chen
193  double hvx, hvz, hvxy, hvxz, hvyz;
194  double h = 1.0 / (1.0 + e);
195  hvx = h * v[0];
196  hvz = h * v[2];
197  hvxy = hvx * v[1];
198  hvxz = hvx * v[2];
199  hvyz = hvz * v[1];
200 
201  matrix_[0][0] = e + hvx * v[0];
202  matrix_[0][1] = hvxy - v[2];
203  matrix_[0][2] = hvxz + v[1];
204 
205  matrix_[1][0] = hvxy + v[2];
206  matrix_[1][1] = e + h * v[1] * v[1];
207  matrix_[1][2] = hvyz - v[0];
208 
209  matrix_[2][0] = hvxz - v[1];
210  matrix_[2][1] = hvyz + v[0];
211  matrix_[2][2] = e + hvz * v[2];
212  }
