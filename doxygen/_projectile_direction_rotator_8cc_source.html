34 void marley::ProjectileDirectionRotator::process_event(marley::Event& ev,
35  marley::Generator& gen)
36 {
37  // First check that the projectile 3-momentum is not a null vector.
38  // If it is, don't bother to rotate coordinates. Also don't bother if
39  // (somehow) the magnitude of the projectile momentum is negative.
40  const auto& projectile = ev.projectile();
41  double pmom = projectile.momentum_magnitude();
42  if ( pmom <= 0. ) return;
43 
44  ThreeVector pdir = { projectile.px() / pmom, projectile.py() / pmom,
45  projectile.pz() / pmom };
46 
47  // If random projectile directions have been requested, then sample
48  // a new one isotropically for this event
49  if ( randomize_projectile_direction_ ) {
50  ThreeVector random_dir = sample_isotropic_direction( gen );
51  this->set_projectile_direction( random_dir );
52  }
53 
54  // If the (unrotated) projectile direction from the event exactly matches the
55  // desired direction, then we can skip the coordinate rotation.
56  if ( pdir == dir_vec_ ) return;
57 
58  // If the (unrotated) projectile direction from the event differs from
59  // the last one that was processed, then we need to recompute the
60  // rotation matrix. Do so and save the unrotated projectile direction
61  // to repeat this check next time.
62  if ( pdir != last_pdir_ ) {
63  last_pdir_ = pdir;
64  rot_matrix_ = marley::RotationMatrix( pdir, dir_vec_ );
65  }
66 
67  // Do the actual rotation of the Particle 3-momenta in the event
68  this->rotate_event( ev );
69 }

71 void marley::ProjectileDirectionRotator::rotate_event( marley::Event& ev ) {
72 
73  // Rotate the initial particles
74  for ( auto* p : ev.get_initial_particles() ) {
75  rot_matrix_.rotate_particle_inplace( *p );
76  }
77 
78  // Rotate the final particles
79  for ( auto* p : ev.get_final_particles() ) {
80  rot_matrix_.rotate_particle_inplace( *p );
81  }

83 }

85 marley::ProjectileDirectionRotator::ThreeVector
86  marley::ProjectileDirectionRotator::sample_isotropic_direction(
87  marley::Generator& gen ) const
88 {
89  // Sample a polar cosine on the interval [-1, 1]
90  double cos_theta = gen.uniform_random_double( -1., 1., true );
91 
92  // Sample an azimuthal angle on the interval [0, 2*pi)
93  double phi = gen.uniform_random_double( 0., marley_utils::two_pi, false );
94 
95  // Compute direction unit vector components
96  double sin_theta = marley_utils::real_sqrt( 1. - std::pow(cos_theta, 2) );
97  double ux = sin_theta * std::cos( phi );
98  double uy = sin_theta * std::sin( phi );
99  double uz = cos_theta;
100 
101  ThreeVector direction = { ux, uy, uz };
102  return direction;
103 }
