50 marley::FermiDiracNeutrinoSource::FermiDiracNeutrinoSource(int particle_id,
51  double Emin, double Emax, double temp, double eta)
52  : NeutrinoSource(particle_id), Emin_(Emin), Emax_(Emax), temperature_(temp),
53  eta_(eta), C_(1.)
54 {
55  // Normalize the source spectrum (not strictly necessary, but having the
56  // spectrum approximately normalized makes the default rejection sampling
57  // tolerance of 1e-8 reliable for finding the maximum of the spectrum)
58  double integral = marley_utils::num_integrate(
59  [this](double E) -> double { return this->pdf(E); }, Emin_, Emax_);
60 
61  // Update the normalization constant, thereby normalizing this object's
62  // pdf in the process.
63  C_ /= integral;
64 }

66 double marley::FermiDiracNeutrinoSource::pdf(double E) const {
67  if (E < Emin_ || E > Emax_) return 0.;
68  else return (C_ / std::pow(temperature_, 4)) * (std::pow(E, 2)
69  / (1 + std::exp((E / temperature_) - eta_)));
70 }

95 double marley::BetaFitNeutrinoSource::pdf(double E) const {
96  if (E < Emin_ || E > Emax_) return 0.;
97  // Guard against NaNs in the std::pow factor below
98  else if ( E == 0. && beta_ < 1. ) return 0.;
99  else return C_ * std::pow(E / Emean_, beta_ - 1.)
100  * std::exp(-beta_ * E / Emean_);
101 }

103 marley::FunctionNeutrinoSource::FunctionNeutrinoSource(int particle_id,
104  double Emin, double Emax, std::function<double(double)> prob_dens_func)
105  : NeutrinoSource(particle_id), Emin_(Emin), Emax_(Emax)
106 {
107  // Normalize the supplied spectrum (not strictly necessary, but having the

109  // tolerance of 1e-8 reliable for finding the maximum of the spectrum)
110  double integral = marley_utils::num_integrate(prob_dens_func, Emin, Emax);
111  probability_density_ = [prob_dens_func, integral](double E)
112  -> double { return prob_dens_func(E) / integral; };

128 double marley::DecayAtRestNeutrinoSource::pdf(double E) const {
129  if (E < Emin_ || E > Emax_) return 0.;
130  // Note that both of these source spectra are normalized to 1
131  // on the energy interval [0., m_mu_ / 2.]
134  int abs_pid = std::abs(pid_);
135  if (abs_pid == marley_utils::ELECTRON_NEUTRINO)
136  return 96. * std::pow(E, 2) * m_mu_to_the_minus_four_
137  * (m_mu_ - 2*E);
138  // Spectrum for muon antineutrinos
139  else return 16. * std::pow(E, 2) * m_mu_to_the_minus_four_
140  * (3*m_mu_ - 4*E);
141 }

145 void marley::GridNeutrinoSource::check_for_errors() {
146  size_t grid_size = grid_.size();
147  if (grid_size < 2) throw marley::Error(std::string("Grid with")
148  + " less than 2 gridpoints passed to the constructor of"
149  + " marley::GridNeutrinoSource.");
150 
151  double sum_of_PDs = 0.;
152  for (size_t j = 0; j < grid_size; ++j) {
153  auto& pair = grid_.at(j);
154  if (pair.first < 0.) throw marley::Error(std::string("All energy")
155  + " values used in a marley::GridNeutrinoSource"
156  + " object must be nonnegative");
157 
158  // Prevent actually sampling an energy value of zero by advancing to
159  // the next representable double value.
160  else if (pair.first == 0.) pair.first = std::nextafter(0.,
161  marley_utils::infinity);
162  if (pair.second < 0.) throw marley::Error(std::string("All PDF")
163  + " values used in a marley::GridNeutrinoSource object must be"
164  + " nonnegative");
165  else sum_of_PDs += pair.second;
166  }
167 
168  if (sum_of_PDs <= 0.) throw marley::Error(std::string("All probability")
169  + " density grid point values are zero for the neutrino source");
170 }
