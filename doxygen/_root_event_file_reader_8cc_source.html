31 bool marley::RootEventFileReader::deduce_file_format() {
32 
33  // First delegate simpler format checks to EventFileReader
34  bool set_format_already = marley::EventFileReader::deduce_file_format();
35  if ( set_format_already ) return true;
36 
37  // Check if the file is in ROOT format. Before doing so, completely suppress
38  // any error messages from ROOT.
39  auto temp_error_level = gErrorIgnoreLevel;
40  gErrorIgnoreLevel = kFatal;
41 
42  tfile_ = std::unique_ptr<TFile>( TFile::Open(file_name_.c_str(), "read") );
43 
44  // We've completed the ROOT format check, so restore the old error messaging
45  // behavior
46  gErrorIgnoreLevel = temp_error_level;
47 
48  // If the file was in ROOT format, then set the format_ data member
49  // accordingly
50  if ( tfile_ ) {
51  format_ = marley::OutputFile::Format::ROOT;
52  return true;
53  }
54  else return false;
55 }

57 void marley::RootEventFileReader::initialize() {
58 
59  if ( format_ == marley::OutputFile::Format::ROOT ) {
60 
61  tfile_->GetObject( "MARLEY_event_tree", ttree_ );
62  if ( !ttree_ ) throw marley::Error("Failed to load MARLEY event TTree"
63  " from the ROOT file \"" + file_name_ + '\"');
64 
65  event_ = std::make_unique<marley::Event>();
66  event_ptr_ = event_.get();
67  ttree_->SetBranchAddress( "event", &event_ptr_ );
68 
69  TParameter<double>* temp_param = nullptr;
70  tfile_->GetObject( "MARLEY_flux_avg_xsec", temp_param );
71  if ( !temp_param ) throw marley::Error("Failed to load the flux-averaged"
72  " total cross section from the ROOT file \"" + file_name_ + '\"');
73 
74  flux_avg_tot_xs_ = temp_param->GetVal();
75 
76  delete temp_param;
77  }
78  else return marley::EventFileReader::initialize();
79 }

81 bool marley::RootEventFileReader::next_event( marley::Event& ev )
82 {
83  this->ensure_initialized();
84 
85  if ( format_ == marley::OutputFile::Format::ROOT ) {
86  ++event_num_;
87  if ( event_num_ < ttree_->GetEntries() ) {
88  ttree_->GetEntry( event_num_ );
89  ev = *event_;
90  return true;
91  }
92 
93  ev = marley::Event();
94  return false;
95  }
96  else return marley::EventFileReader::next_event( ev );
97 }
