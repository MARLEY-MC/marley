52 void marley::Target::initialize( const std::vector<marley::TargetAtom>& nuclides,
53  const std::vector<double>& atom_fracs )
54 {
55  // Check that we have the same number of target atoms and atom fractions
56  if ( nuclides.size() != atom_fracs.size() ) {
57  throw marley::Error("Different numbers of target atoms and atom fractions"
58  " encountered in the constructor of marley::Target");
59  }
60 
61  // Check that we have at least one target nuclide. Note that the vector's
62  // size is unsigned and therefore can't go negative.
63  if ( nuclides.size() == 0u ) throw marley::Error( "No target atoms"
64  " specified when constructing a marley::Target object" );
65 
66  // Make sure the atom fractions sum to unity by explicitly renormalizing them
67  double sum = 0.;
68  for ( size_t j = 0u; j < atom_fracs.size(); ++j ) {
69  double af = atom_fracs.at( j );
70  const auto& nuc = nuclides.at( j );
71  if ( af < 0. ) throw marley::Error("Invalid atom fraction "
72  + std::to_string(af) + " encountered for a "
73  + nuc.to_string() + " target atom");
74  else if ( af == 0. ) MARLEY_LOG_WARNING() << "Atom fraction of zero"
75  << " encountered for a " << nuc << " target atom";
76  sum += af;
77  }
78 
79  // If the sum of the fractions is non-positive, we'll have trouble.
80  if ( sum <= 0. ) throw marley::Error( "Sum of atom fractions = "
81  + std::to_string(sum) + " in the constructor of marley::Target" );
82 
83  std::vector<double> renorm_fracs;
84  for ( const auto& af : atom_fracs ) renorm_fracs.push_back( af / sum );
85 
86  // We're ready. Initialize the target atom map.
87  for ( size_t j = 0u; j < nuclides.size(); ++j ) {
88  const auto& nuc = nuclides.at( j );
89  const auto& frac = renorm_fracs.at( j );
90  // If we have duplicate nuclides, throw an error (likely to be a mistake)
91  if ( atom_fractions_.count(nuc) ) throw marley::Error("Duplicate atom"
92  " fractions specified for the target atom " + nuc.to_string() + " in"
93  " the constructor of marley::Target");
94  atom_fractions_[ nuc ] = frac;
95  }
96 }

98 double marley::Target::atom_fraction( const marley::TargetAtom& atom ) const {
99  // If we have an atom fraction stored in the map for this nuclide,
100  // then just return it
101  auto it = atom_fractions_.find( atom );
102  if ( it != atom_fractions_.end() ) return it->second;
103  // If we couldn't find it, then just return zero
104  return 0.;
105 }

111 void marley::Target::print( std::ostream& out ) const {
112  size_t num_atom_types = atom_fractions_.size();
113  size_t count = 1;
114  for ( const auto& pair : atom_fractions_ ) {
115  const auto& nuc = pair.first;
116  const auto& frac = pair.second;
117  std::string A_str = std::to_string( nuc.A() );
118  marley_utils::pad_left_inplace( A_str, 3 );
119 
120  auto it = marley_utils::element_symbols.find( nuc.Z() );
121  std::string symb = "??";
122  if ( it != marley_utils::element_symbols.end() ) symb = it->second;
123  marley_utils::pad_right_inplace( symb, 2 );
124 
125  out << A_str << symb << " = " << frac;
126  // Add a newline to all but the last entry
127  if ( count < num_atom_types ) out << '\n';
128  ++count;
129  }
130 }
