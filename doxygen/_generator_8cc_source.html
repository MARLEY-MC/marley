67 marley::Event marley::Generator::create_event() {
68 
69  // (1) Select a reacting neutrino energy and reaction using the
70  // flux-weighted total cross section(s)
71  double E_nu;
72  marley::Reaction& r = sample_reaction( E_nu );
73 
74  // (2) Create the prompt two-two scattering event using the
75  // sampled reaction object
76  marley::Event ev = r.create_event( source_->get_pid(), E_nu, *this );
77 
78  // (3) If needed, de-excite the final-state residue
79  if ( do_deexcitations_ ) {
80  marley::NucleusDecayer nd;
81  nd.process_event( ev, *this );
82  }
83 
84  // (4) If needed, rotate the event to match the desired projectile direction
85  rotator_.process_event( ev, *this );
86 
87  // Return the completed event object
88  return ev;
89 }

116 void marley::Generator::normalize_E_pdf() {
117 
118  // If we're not ready to do the normalization, then just return without doing
119  // anything. JSONConfig may set this flag to prevent premature calls to
120  // normalize_E_pdf() as the Generator is being constructed
121  if ( dont_normalize_E_pdf_ ) return;
122 
123  // This function is called whenever the reacting neutrino energy PDF changes,
124  // so reset the estimated maximum PDF value to its default. We will update
125  // this during rejection sampling.
126  E_pdf_max_ = E_PDF_MAX_DEFAULT_;
127 
128  // Treat monoenergetic sources differently since they can cause
129  // problems for the standard numerical integration check
130  if ( source_->get_Emin() == source_->get_Emax() ) {
131  // Set the normalization factor back to one. It's used
132  // in the call to E_pdf() below, so we need to do this before
133  // we assign it a different value.
134  norm_ = 1.0; //
135  // Now norm_ is assigned to be the product of the total cross section times
136  // the source PDF at energy Emin
137  norm_ = E_pdf( source_->get_Emin() );
138  if ( norm_ <= 0. || std::isnan(norm_) ) {
139  throw marley::Error("The total cross section for all defined reactions"
140  " is <= 0 or NaN for the neutrino energy defined in a monoenergetic"
141  " source. Please verify that your neutrino source produces particles"
142  " above threshold for at least one reaction.");
143  }
144  }
145  else {
146  // Reset the normalization factor to its default of one until
147  // we can calculate the new value. This prevents strange things
148  // from happening when we lose precision due to an abnormally
149  // high or low norm_ value from a previous source or reaction
150  // definition.
151  norm_ = 1.;
152 
153  // Update the normalization factor for use with the reacting neutrino
154  // energy probability density function
155  norm_ = marley_utils::num_integrate( [this](double E)
156  -> double { return this->E_pdf(E); }, source_->get_Emin(),
157  source_->get_Emax() );
158 
159  if ( norm_ <= 0. || std::isnan(norm_) ) {
160  throw marley::Error( "The integral of the cross-section-weighted"
161  " neutrino flux is <= 0 or NaN. Please verify that your neutrino"
162  " source spectrum produces significant flux above the reaction"
163  " threshold(s)." );
164  }
165  }
166 }

168 // Sample a random double uniformly between min and max using the class
169 // member random number generator rand_gen_. The inclusive flag
170 // determines whether or not max is included in the range. That is,
171 // when inclusive == false, the sampling is done on the interval [min, max),
172 // while inclusive == true uses [min, max].
173 double marley::Generator::uniform_random_double(double min, double max,
174  bool inclusive)
175 {
176  // Defaults to sampling from [0,1). We will always
177  // explicitly supply the upper and lower bounds to
178  // this distribution, so we won't worry about the
179  // default setting.
180  static std::uniform_real_distribution<double> udist;
181 
182  double max_to_use;
183 
184  if ( inclusive ) { // sample from [min, max]
185 
186  // Find the double value that comes immediately after max. This allows us
187  // to sample uniformly on [min, max] rather than [min,max). This trick
188  // comes from http://tinyurl.com/n3ocg3p.
189  max_to_use = std::nextafter( max, std::numeric_limits<double>::max() );
190  }
191  else { // sample from [min, max)
192  max_to_use = max;
193  }
194 
195  std::uniform_real_distribution<double>::param_type params( min, max_to_use );
196 
197  // Sample a random double from this distribution
198  return udist( rand_gen_, params );
199 }

219 double marley::Generator::rejection_sample(const std::function<double(double)>& f,
220  double xmin, double xmax, double& fmax, double safety_factor,
221  double max_search_tolerance)
222 {
223  // If we were passed the value marley_utils::UNKNOWN_MAX for fmax, then this
224  // signals that we need to search for the function maximum ourselves.
225  // Otherwise, we'll assume that the value passed over is good.
226  if ( fmax == marley_utils::UNKNOWN_MAX ) {
227  // This variable will be loaded with the value of x
228  // that corresponds to the maximum of f(x).
229  // We don't actually use this, but currently it's
230  // a required parameter of marley_utils::maximize
231  double x_at_max;
232 
233  // Maximize the function and multiply by a safety factor just
234  // in case we didn't quite find the exact peak
235  fmax = marley_utils::maximize(f, xmin, xmax, max_search_tolerance,
236  x_at_max) * safety_factor;
237  }
238 
239  double x, y, val;
240 
241  do {
242  // Sample x value uniformly from [xmin, xmax]
243  x = uniform_random_double(xmin, xmax, true);
244 
245  // Sample y uniformly from [0, fmax]
246  y = uniform_random_double(0, fmax, true);
247 
248  val = f(x);
249  if ( val > fmax ) {
250 
251  MARLEY_LOG_WARNING() << "PDF value f(x) = "
252  << val << " at x = " << x << " exceeded the estimated maximum"
253  << " fmax = " << fmax << " during rejection sampling.";
254 
255  fmax = val * safety_factor;
256  MARLEY_LOG_WARNING() << "A new estimate fmax = " << val * safety_factor
257  << " will now be adopted.";
258  }
259  }
260  // Keep sampling until you get a y value less than f(x)
261  // (the probability density function evaluated at the sampled value of x)
262  while ( y > val );
263 
264  return x;
265 }

267 double marley::Generator::E_pdf(double E) {
268 
269  // Initialize the return value to zero
270  double pdf = 0.;
271 
272  // Sum all of the reaction total cross sections, saving
273  // each individual value along the way. Take weighting
274  // by atom fraction in the target material into account.
275  for ( size_t j = 0, s = reactions_.size(); j < s; ++j ) {
276 
277  // Get the current reaction
278  const auto& react = reactions_.at( j );
279 
280  // Compute the total cross section for the current reaction for a single
281  // target atom
282  double tot_xs = react->total_xs( source_->get_pid(), E );
283 
284  // If the target_ member has not been initialized, don't bother doing any
285  // weighting by atom fraction (equivalent to a weight of unity for all
286  // target atoms)
287  if ( target_ ) {
288  // If it has been configured, then apply the appropriate atom fraction
289  // weight from the target as appropriate.
290  tot_xs *= target_->atom_fraction( react->atomic_target() );
291  }
292 
293  // Cache the atom-fraction-weighted total cross section for sampling a
294  // reaction mode later
295  total_xs_values_.at( j ) = tot_xs;
296 
297  // Add the weighted total cross section value to the total
298  pdf += tot_xs;
299  }
300 
301  // Normally, we want to fold the flux with the reaction cross section(s)
302  // in order to obtain the distribution of reacting neutrino energies
303  if ( weight_flux_ ) {
304  // Multiply the total cross section by the neutrino spectrum
305  // from the source object to get the (unnormalized) PDF
306  // for sampling reacting neutrino energies.
307  pdf *= source_->pdf(E);
308  }
309  else {
310  // If the user has specifically requested it, don't weight the
311  // energy PDF by the cross section(s), as long as at least one of them
312  // is non-vanishing
313  if ( pdf <= 0. ) return 0.;
314  pdf = source_->pdf(E);
315  }
316 
317  // Divide by the normalization factor (computed when this source
318  // was made available to the Generator) to obtain the normalized PDF.
319  return pdf / norm_;
320 }

322 marley::Reaction& marley::Generator::sample_reaction(double& E) {
323  if ( reactions_.empty() ) throw marley::Error("Cannot sample"
324  " a reaction in marley::Generator::sample_reaction(). The vector of"
325  " marley::Reaction objects owned by this generator is empty.");
326 
327  // Store the "old" value of E_pdf_max_, i.e., the one we had before calling
328  // rejection_sample(). This will be used to check for problems.
329  double old_max = E_pdf_max_;
330 
331  // TODO: protect against source_ changing E_min or E_max after you compute
332  // the normalization factor norm_ in marley::Generator::init()
333  E = rejection_sample([this](double E_nu)
334  -> double { return this->E_pdf(E_nu); }, source_->get_Emin(),
335  source_->get_Emax(), E_pdf_max_);
336 
337  // If the value of max changed after the call to rejection_sample() and the
338  // old value wasn't UNKNOWN_MAX, then the rejection sampling routine must
339  // have encountered a PDF value that was larger than our estimated maximum.
340  // Alert the user about this and advise them to change the configuration
341  // appropriately to avoid a biased reacting neutrino energy distribution.
342  static bool issued_long_error_message = false;
343  if ( old_max != marley_utils::UNKNOWN_MAX
344  && old_max != E_pdf_max_ )
345  {
346  if ( !issued_long_error_message ) {
347  MARLEY_LOG_ERROR() << "Estimation of the maximum PDF value failed when"
348  << " using a rejection method to sample reacting neutrino energies.\n"
349  << "This may occur when, e.g., an incident neutrino flux"
350  << " is used that includes multiple sharp peaks.\n"
351  << "To avoid biasing the energy distribution, please rerun the simulation"
352  << " after adding the following line to the MARLEY job configuration"
353  << " file:\n"
354  << " energy_pdf_max: " << E_pdf_max_ << ",\n"
355  << "If this error message persists after raising energy_pdf_max to a"
356  << " relatively high value, please contact the MARLEY developers for"
357  << " troubleshooting help.";
358  issued_long_error_message = true;
359  }
360  else {
361  MARLEY_LOG_ERROR() << "The maximum PDF value for sampling reacting neutrino"
362  << " energies was exceeded again. The new estimated maximum is"
363  << "\n energy_pdf_max: " << E_pdf_max_ << ',';
364  }
365  }
366 
367  // The atom-fraction-weighted total cross section values have already been
368  // updated by the final call to E_pdf() during rejection sampling, so we can
369  // now sample a reaction type using our discrete distribution object.
370  std::discrete_distribution<size_t>::param_type
371  params( total_xs_values_.begin(), total_xs_values_.end() );
372  size_t r_index = r_index_dist_( rand_gen_, params );
373  return *reactions_.at( r_index );
374 }

462 double marley::Generator::inverse_transform_sample(
463  const std::function<double(double)>& f, double xmin, double xmax,
464  double bisection_tolerance)
465 {
466  // Build an approximate CDF corresponding to the integral of the input PDF.
467  // Use a polynomial approximant at Chebyshev points to do it.
469  marley::ChebyshevInterpolatingFunction func(f, xmin, xmax,
470  DEFAULT_N_CHEBYSHEV);
471  auto cdf = func.cdf();
472 
473  // Now that we have a CDF to use for sampling, delegate the rest of the
474  // action to the overloaded version of this function.
475  return this->inverse_transform_sample(cdf, xmin, xmax, bisection_tolerance);
476 }

478 double marley::Generator::inverse_transform_sample(
479  const marley::ChebyshevInterpolatingFunction& cdf, double xmin, double xmax,
480  double bisection_tolerance)
481 {
482  // Sample a probability value uniformly on [0, 1]
483  double prob = uniform_random_double(0., 1., true);
484 
485  // If we chose an endpoint, we're done, so just return the appropriate one
486  if ( prob == 0. ) return xmin;
487  else if ( prob == 1. ) return xmax;
488 
489  // A properly normalized CDF should evaluate to unity at x = xmax. We enforce
490  // this here so that the user doesn't have to do it in advance.
491  double norm = cdf.evaluate( xmax );
492 
493  // Find the x value corresponding to the sampled probability via bisection
494  // (slow but robust)
495  double a = xmin;
496  double b = xmax;
497  while ( (b - a) > bisection_tolerance ) {
498  double midpoint = (a + b) / 2.;
499  double mid_cdf = cdf.evaluate( midpoint ) / norm;
500  // If the CDF at the midpoint exactly matches our sampled
501  // probability, we're done. Just return the midpoint.
502  if ( mid_cdf == prob ) return midpoint;
503  // Otherwise, shrink the bisection interval and try again
504  else if ( mid_cdf > prob ) b = midpoint;
505  else a = midpoint; // mid_cdf < prob
506  }
507 
508  // Return the midpoint of the bisection interval as our sampled x value
509  double x = (a + b) / 2.;
510  return x;
511 }

513 double marley::Generator::flux_averaged_total_xs() const {
514  // If we've disabled weighting the neutrino energy PDF
515  // by the total cross section, just return zero
516  if ( !weight_flux_ ) return 0.;
517 
518  double avg_total_xs = 0.;
519 
520  // For a monoenergetic source, don't bother to do the full
521  // integral
522  double Emin = source_->get_Emin();
523  double Emax = source_->get_Emax();
524  if ( Emin == Emax ) {
525  avg_total_xs = norm_ / source_->pdf( Emin );
526  }
527  else {
528  double source_norm = marley_utils::num_integrate(
529  [this](double Ev) -> double { return this->source_->pdf(Ev); },
530  Emin, Emax);
531 
532  // Use the precomputed integral of the reacting neutrino energy PDF
533  avg_total_xs = norm_ / source_norm;
534  }
535  return avg_total_xs;
536 }

560 double marley::Generator::total_xs( int pdg_a, double KEa, int pdg_atom,
561  std::vector<size_t>* index_vec, std::vector<double>* xsec_vec ) const
562 {
563  double xsec_sum = 0.;
564 
565  if ( index_vec ) index_vec->clear();
566  if ( xsec_vec ) xsec_vec->clear();
567 
568  for ( size_t j = 0u; j < reactions_.size(); ++j ) {
569 
570  const auto& r = reactions_.at( j );
571 
572  // Skip reactions which involve a different target atom
573  if ( pdg_atom != r->atomic_target().pdg() ) continue;
574 
575  // Skip reactions which involve a different projectile
576  if ( pdg_a != r->pdg_a() ) continue;
577 
578  // If the cross section is non-vanishing, store information about it
579  // (as appropriate) and add it to the sum
580  double xsec = r->total_xs( pdg_a, KEa );
581  if ( xsec > 0. ) {
582  xsec_sum += xsec;
583  if ( index_vec ) index_vec->push_back( j );
584  if ( xsec_vec ) xsec_vec->push_back( xsec );
585  }
586 
587  }
588 
589  return xsec_sum;
590 }

593 marley::Event marley::Generator::create_event( int pdg_a, double KEa,
594  int pdg_atom, const std::array<double, 3>& dir_vec )
595 {
596  // (1) Sample a reaction mode from all configured reactions that can handle
597  // the given initial-state parameters
598  std::vector<size_t> indices;
599  std::vector<double> xsecs;
600  double tot_xsec = this->total_xs( pdg_a, KEa, pdg_atom, &indices, &xsecs );
601 
602  if ( xsecs.empty() || tot_xsec <= 0. ) throw marley::Error(
603  "Cannot create an event for a projectile with kinetic energy = "
604  + std::to_string(KEa) + " MeV and PDG code " + std::to_string(pdg_a)
605  + " striking an atom with PDG code " + std::to_string(pdg_atom)
606  + ". The total cross section vanishes for all configured reactions" );
607 
608  // The total cross section values and indices in the full reactions_ vector
609  // have already been loaded into temporary vectors, so we can immediately use
610  // those to sample a reaction using a discrete distribution.
611  std::discrete_distribution<size_t> react_dist( xsecs.begin(), xsecs.end() );
612  size_t sampled_index = react_dist( rand_gen_ );
613  auto& r = reactions_.at( indices.at(sampled_index) );
614 
615  // (2) Create the prompt two-two scattering event using the sampled reaction
616  // object
617  marley::Event ev = r->create_event( pdg_a, KEa, *this );
618 
619  // Do the usual post-processing
620 
621  // (3) If needed, de-excite the final-state residue
622  if ( do_deexcitations_ ) {
623  marley::NucleusDecayer nd;
624  nd.process_event( ev, *this );
625  }
626 
627  // (4) If needed, rotate the event to match the desired projectile direction
628  // Set the incident neutrino direction for this event
629  static marley::ProjectileDirectionRotator my_rotator;
630  my_rotator.set_projectile_direction( dir_vec );
631 
632  // Rotate the coordinate system of the event if needed
633  my_rotator.process_event( ev, *this );
634 
635  // Return the completed event object
636  return ev;
637 }

639 // Compute the abundance-weighted total reaction cross section at fixed
640 // projectile kinetic energy
641 double marley::Generator::total_xs( int pdg_a, double KEa ) const {
642
