53  // Temporary storage for output TTree branch variables
54  double Ex; // nuclear excitation energy
55  int twoJ; // two times the residue spin immediately after the two-two
56  // scattering reaction
57  int par; // integer representation of the intrinsic parity of the
58  // residue immediately following the two-two reaction
59  double flux_avg_tot_xsec; // flux-averaged total cross section
60  double Ev, KEv, pxv, pyv, pzv; // projectile
61  double Mt; // target mass
62  double El, KEl, pxl, pyl, pzl; // ejectile
63  double Er, KEr, pxr, pyr, pzr; // residue (after de-excitations)
64  int pdgv, pdgt, pdgl, pdgr; // PDG codes
65  int np; // number of de-excitation products (final-state particles other
66  // than the ejectile and residue)
67 
68  // Inpormation about each of the other final-state particles
69  std::vector<int> PDGs;
70  std::vector<double> Es, KEs, pXs, pYs, pZs;
71 
72  // Check whether the output file exists and warn the user before
73  // overwriting it if it does
74  std::ifstream temp_stream( argv[1] );
75  if ( temp_stream ) {
76  bool overwrite = marley_utils::prompt_yes_no(
77  "Really overwrite " + std::string(argv[1]) + '?');
78  if ( !overwrite ) {
79  std::cout << "Action aborted.\n";
80  return 0;
81  }
82  }
83 
84  TFile out_tfile( argv[1], "recreate" );
85  TTree* out_tree = new TTree("mst", "MARLEY summary tree");
86 
87  // projectile branches
88  out_tree->Branch("pdgv", &pdgv, "pdgv/I");
89  out_tree->Branch("Ev", &Ev, "Ev/D");
90  out_tree->Branch("KEv", &KEv, "KEv/D");
91  out_tree->Branch("pxv", &pxv, "pxv/D");
92  out_tree->Branch("pyv", &pyv, "pyv/D");
93  out_tree->Branch("pzv", &pzv, "pzv/D");
94 
95  // target branches
96  out_tree->Branch("pdgt", &pdgt, "pdgt/I");
97  out_tree->Branch("Mt", &Mt, "Mt/D");
98 
99  // ejectile branches
100  out_tree->Branch("pdgl", &pdgl, "pdgl/I");
101  out_tree->Branch("El", &El, "El/D");
102  out_tree->Branch("KEl", &KEl, "KEl/D");
103  out_tree->Branch("pxl", &pxl, "pxl/D");
104  out_tree->Branch("pyl", &pyl, "pyl/D");
105  out_tree->Branch("pzl", &pzl, "pzl/D");
106 
107  // residue branches
108  out_tree->Branch("pdgr", &pdgr, "pdgr/I");
109  out_tree->Branch("Er", &Er, "Er/D");
110  out_tree->Branch("KEr", &KEr, "KEr/D");
111  out_tree->Branch("pxr", &pxr, "pxr/D");
112  out_tree->Branch("pyr", &pyr, "pyr/D");
113  out_tree->Branch("pzr", &pzr, "pzr/D");
114 
115  // Nuclear excitation energy branch
116  out_tree->Branch("Ex", &Ex, "Ex/D");
117 
118  // Spin and parity branches
119  out_tree->Branch("twoJ", &twoJ, "twoJ/I");
120  out_tree->Branch("parity", &par, "parity/I");
121 
122  // De-excitation products (final-state particles other than the
123  // ejectile and ground-state residue)
124  out_tree->Branch("np", &np, "np/I");
125  out_tree->Branch("pdgp", PDGs.data(), "pdgp[np]/I");
126  out_tree->Branch("Ep", Es.data(), "Ep[np]/D");
127  out_tree->Branch("KEp", KEs.data(), "KEp[np]/D");
128  out_tree->Branch("pxp", pXs.data(), "pxp[np]/D");
129  out_tree->Branch("pyp", pYs.data(), "pyp[np]/D");
130  out_tree->Branch("pzp", pZs.data(), "pzp[np]/D");
131 
132  // Flux-averaged total cross section
133  out_tree->Branch("xsec", &flux_avg_tot_xsec, "xsec/D");
134 
135  // Prepare to read the input file(s)
136  std::vector<std::string> input_file_names;
137  for ( int i = 2; i < argc; ++i ) input_file_names.push_back( argv[i] );
138 
139  // File loop
140  for ( const auto& file_name : input_file_names ) {
141 
142  // Open the current file for reading
143  marley::RootEventFileReader refr( file_name );
144  std::cout << "Opened file \"" << file_name << "\"\n";
145 
146  // Temporary object to use for reading in saved events
147  marley::Event ev;
148 
149  // Event loop
150  int event_num = 0;
151  while ( refr >> ev ) {
152

154 
155  PDGs.clear();
156  Es.clear();
157  KEs.clear();
158  pXs.clear();
159  pYs.clear();
160  pZs.clear();
161 
162  pdgv = ev.projectile().pdg_code();
163  Ev = ev.projectile().total_energy();
164  KEv = ev.projectile().kinetic_energy();
165  pxv = ev.projectile().px();
166  pyv = ev.projectile().py();
167  pzv = ev.projectile().pz();
168 
169  pdgt = ev.target().pdg_code();
170  Mt = ev.target().mass();
171 
172  pdgl = ev.ejectile().pdg_code();
173  El = ev.ejectile().total_energy();
174  KEl = ev.ejectile().kinetic_energy();
175  pxl = ev.ejectile().px();
176  pyl = ev.ejectile().py();
177  pzl = ev.ejectile().pz();
178 
179  pdgr = ev.residue().pdg_code();
180  Er = ev.residue().total_energy();
181  KEr = ev.residue().kinetic_energy();
182  pxr = ev.residue().px();
183  pyr = ev.residue().py();
184  pzr = ev.residue().pz();
185 
186  Ex = ev.Ex();
187  twoJ = ev.twoJ();
188  par = static_cast<int>( ev.parity() );
189  flux_avg_tot_xsec = refr.flux_averaged_xsec();
190 
191  const auto& fparts = ev.get_final_particles();
192  np = fparts.size() - FIRST_PROD_IDX;
193  for ( size_t j = FIRST_PROD_IDX; j < fparts.size(); ++j ) {
194  const auto* fp = fparts.at( j );
195  PDGs.push_back( fp->pdg_code() );
196  Es.push_back( fp->total_energy() );
197  KEs.push_back( fp->kinetic_energy() );
198  pXs.push_back( fp->px() );
199  pYs.push_back( fp->py() );
200  pZs.push_back( fp->pz() );
201  }
202 
203  // Update the branch addresses (manipulating the vectors may have
204  // invalidated them)
205  out_tree->SetBranchAddress("pdgp", PDGs.data());
206  out_tree->SetBranchAddress("Ep", Es.data());
207  out_tree->SetBranchAddress("KEp", KEs.data());
208  out_tree->SetBranchAddress("pxp", pXs.data());
209  out_tree->SetBranchAddress("pyp", pYs.data());
210  out_tree->SetBranchAddress("pzp", pZs.data());
211 
212  out_tree->Fill();
213 
214  ++event_num;
215  } // event loop
216  } // file loop
217 
218  out_tfile.cd();
219  out_tree->Write();
220  out_tfile.Close();
221  return 0;
222 }
