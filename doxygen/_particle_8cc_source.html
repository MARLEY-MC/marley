27  double read_JSON_double( const std::string& key, const marley::JSON& json ) {
28 
29  if ( !json.has_key(key) ) throw marley::Error("Missing key " + key
30  + " encountered in an input JSON particle object");
31 
32  bool ok = false;
33  const auto& temp = json.at( key );
34  double result = temp.to_double( ok );
35  if ( !ok ) throw marley::Error("Invalid value " + temp.to_string()
36  + " associated with the " + key + " key encountered in an input"
37  " JSON particle object");
38  return result;
39  }
