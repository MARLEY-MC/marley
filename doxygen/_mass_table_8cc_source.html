83 double marley::MassTable::get_particle_mass(int particle_id) const {
84  int id = particle_id;
85  // The lookup table only includes entries for particles (as opposed to
86  // antiparticles), so flip the sign of the input particle id for the
87  // lookup if it represents an antiparticle.
88  if (id < 0) id *= -1;
89  // Find the particle's mass in the lookup table, and convert its
90  // value from micro-amu to MeV
91  return micro_amu_ * particle_masses_.at(id);
92 }

103 double marley::MassTable::lookup_atomic_mass(int nucleus_pid, bool& exp,
104  bool theory_ok) const
105 {
106  // Find the atom's mass (in micro-amu) in the lookup table using its
107  // nucleus's particle ID number. If it can't be found, either return a
108  // theoretical mass using the liquid drop model or throw an error.
109  // std::unordered_map<int, double>::iterator search
110  auto search = atomic_masses_.find(nucleus_pid);
111 
112  // If the mass was found in the lookup table, return it and flag it as an
113  // experimental value
114  if (search != atomic_masses_.end()) {
115  exp = true;
116  return search->second;
117  }
118  // Otherwise, return a theoretical estimate using the liquid drop model or
119  // throw an error depending on whether the user has indicated that using a
120  // theoretical estimate is acceptable. For either case, set the experimental
121  // flag to false.
122  else {
123  exp = false;
124 
125  int Z = marley_utils::get_particle_Z(nucleus_pid);
126  int A = marley_utils::get_particle_A(nucleus_pid);
127 
128  if (theory_ok) {
129  return liquid_drop_model_atomic_mass(Z, A);
130  }
131  else throw marley::Error(std::string("Entry for Z = ")
132  + std::to_string(Z) + " and A = " + std::to_string(A)
133  + " not found in the MARLEY atomic mass table.");
134  }
135 }

166 double marley::MassTable::get_binding_energy(int Z, int A, bool theory_ok) const
167 {
168  int N = A - Z;
169  double m_hydrogen_1 = atomic_masses_.at(1000010010);
170  double mn = particle_masses_.at(marley_utils::NEUTRON);
171 
172  bool exp;
173  double mN = lookup_atomic_mass(Z, A, exp, theory_ok);
174 
175  // Experimental masses are given in micro-amu, while our liquid drop model
176  // estimates are given in MeV, so adjust the calculation appropriately
177  // depending on which unit we are using for mN.
178  if (exp) return micro_amu_ * (Z*m_hydrogen_1 + N*mn - mN);
179  else return micro_amu_ * (Z*m_hydrogen_1 + N*mn) - mN;
180 }

205 double marley::MassTable::get_fragment_separation_energy(int Z, int A, int pid,
206  bool theory_ok) const
207 {
208  int Zx = marley_utils::get_particle_Z(pid);
209  int Zf = Z - Zx;
210  int Af = A - marley_utils::get_particle_A(pid);
211 
212  double extra_mass = Zx*particle_masses_.at(marley_utils::ELECTRON)
213  + particle_masses_.at(pid);
214 
215  bool exp_i, exp_f;
216  double m_atom_initial = lookup_atomic_mass(Z, A, exp_i, theory_ok);
217  double m_atom_final = lookup_atomic_mass(Zf, Af, exp_f, theory_ok);
218 
219  if (exp_i) {
220  if (exp_f) return micro_amu_*(m_atom_final - m_atom_initial + extra_mass);
221  else return m_atom_final + micro_amu_*(extra_mass - m_atom_initial);
222  }
223  else if (exp_f) return micro_amu_*(m_atom_final + extra_mass)
224  - m_atom_initial;
225  else return micro_amu_*extra_mass + m_atom_final - m_atom_initial;
226 }

232 double marley::MassTable::liquid_drop_model_mass_excess(int Z, int A) const {
233 
234  // Liquid drop model parameters (taken from paper by A. J. Koning, et al.)
235  static constexpr double Mn = 8.07144; // MeV
236  static constexpr double MH = 7.28899; // MeV
237  static constexpr double a1 = 15.677; // MeV
238  static constexpr double a2 = 18.56; // MeV
239  static constexpr double kappa = 1.79;
240  static constexpr double c3 = 0.717; // MeV
241  static constexpr double c4 = 1.21129; // MeV
242 
243  int N = A - Z;
244 
245  double kappa_term = kappa * std::pow((N - Z) / static_cast<double>(A), 2);
246  double c1 = a1 * (1 - kappa_term);
247  double c2 = a2 * (1 - kappa_term);
248 
249  double Evol = -c1 * A;
250  double Esur = c2 * std::pow(A, 2.0/3.0);
251  double Ecoul = (c3 / std::pow(A, 1.0/3.0) - c4 / A) * std::pow(Z, 2);
252 
253  double delta_LDM = 0;
254  bool z_odd = Z % 2;
255  bool n_odd = N % 2;
256 
257  // delta_LDM will be zero if the nucleus is odd-even
258  if (z_odd && n_odd) delta_LDM = 11/std::sqrt(A);
259  else if (!z_odd && !n_odd) delta_LDM = -11/std::sqrt(A);
260 
261  return Mn * N + MH * Z + Evol + Esur + Ecoul + delta_LDM;
262 }

272 // Returns the smallest nuclear fragment emission threshold (separation energy)
273 // for a particular initial nucleus. All nuclear fragments recognized by the
274 // StructureDatabase class are considered.
275 double marley::MassTable::unbound_threshold(const int Zi, const int Ai) const
276 {
277  // Before looking up the separation energies, start by setting the unbound
278  // threshold to infinity.
279  double unbound_threshold = std::numeric_limits<double>::max();
280  MARLEY_LOG_DEBUG() << "unbound_threshold = " << unbound_threshold << '\n';
281 
282  // Loop over each available nuclear fragment. If it has a smaller separation
283  // energy than the current value of unbound_threshold, update the stored value
284  for ( const auto& pair : marley::StructureDatabase::fragments() ) {
285  const marley::Fragment& f = pair.second;
286  double thresh = this->fragment_emission_threshold( Zi, Ai, f );
287  MARLEY_LOG_DEBUG() << f.get_pid() << " emission threshold = "
288  << thresh << '\n';
289  if ( thresh < unbound_threshold ) unbound_threshold = thresh;
290  MARLEY_LOG_DEBUG() << "unbound_threshold = " << unbound_threshold << '\n';
291  }
292 
293  return unbound_threshold;
294 }

303 void marley::MassTable::assign_masses(const marley::JSON& obj_array,
304  const std::string& array_key, std::unordered_map<int, double>& map_to_use)
305 {
306  if ( !obj_array.is_array() ) {
307  throw marley::Error("The \"" + array_key + "\" key in the"
308  " mass data file " + data_file_name_
309  + " does not refer to a JSON array.");
310  }
311  auto elements = obj_array.array_range();
312  if ( elements.begin() == elements.end() ) {
313  throw marley::Error("The \"" + array_key + "\" array in"
314  " the mass data file " + data_file_name_ + " is empty.");
315  }
316  bool pdg_ok, mass_ok;
317  for (const auto& el : elements) {
318 
319  if ( !el.has_key("pdg") ) {
320  throw marley::Error("Missing pdg code for an element of the"
321  " \"" + array_key + "\" array in the mass data file " + data_file_name_);
322  }
323  else if ( !el.has_key("mass") ) {
324  throw marley::Error("Missing mass for an element of the"
325  " \"" + array_key + "\" array in the mass data file " + data_file_name_);
326  }
327 
328  auto pdg_json = el.at("pdg");
329  int pdg = pdg_json.to_long( pdg_ok );
330  if ( !pdg_ok ) throw marley::Error(std::string("Invalid PDG code \"")
331  + pdg_json.dump_string() + "\" given in the \"" + array_key + "\" array"
332  " in the mass data file " + data_file_name_);
333 
334  auto mass_json = el.at("mass");
335  double mass = mass_json.to_double( mass_ok );
336  if ( !mass_ok ) throw marley::Error(std::string("Invalid mass value \"")
337  + mass_json.dump_string() + "\" given in the \"" + array_key + "\" array"
338  " in the mass data file " + data_file_name_);
339 
340  map_to_use[ pdg ] = mass;
341  }
