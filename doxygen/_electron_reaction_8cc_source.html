59 void marley::ElectronReaction::set_coupling_constants()
60 {
61  if (pdg_a_ == marley_utils::ELECTRON_NEUTRINO) {
62  g1_ = marley_utils::ONE_HALF + marley_utils::sin2thetaw;
63  g2_ = marley_utils::sin2thetaw;
64  }
65  else if (pdg_a_ == marley_utils::ELECTRON_ANTINEUTRINO) {
66  g1_ = marley_utils::sin2thetaw;
67  g2_ = marley_utils::ONE_HALF + marley_utils::sin2thetaw;
68  }
69  else if (pdg_a_ == marley_utils::MUON_NEUTRINO ||
70  pdg_a_ == marley_utils::TAU_NEUTRINO)
71  {
72  g1_ = -marley_utils::ONE_HALF + marley_utils::sin2thetaw;
73  g2_ = marley_utils::sin2thetaw;
74  }
75  else if (pdg_a_ == marley_utils::MUON_ANTINEUTRINO ||
76  pdg_a_ == marley_utils::TAU_ANTINEUTRINO)
77  {
78  g1_ = marley_utils::sin2thetaw;
79  g2_ = -marley_utils::ONE_HALF + marley_utils::sin2thetaw;

81  else throw marley::Error("Unrecognized projectile PDG code "
82  + std::to_string(pdg_a_) + " encountered in marley::Electron"

86 double marley::ElectronReaction::total_xs(int pdg_a, double KEa) const {
87 
88  // If the cross section was requested for a different projectile,
89  // then just return zero.
90  if ( pdg_a != pdg_a_ ) return 0.;
91 
92  // If we're below threshold, then just return zero
93  if ( KEa < KEa_threshold_ ) return 0.;
94 
95  // Mandelstam s (square of the total center of momentum frame energy)
96  double s = std::pow(ma_ + mb_, 2) + 2.*mb_*KEa;
97 
98  // CM frame ejectile total energy
99  double Ec_cm = (s + mc_*mc_ - md_*md_) / ( 2. * std::sqrt(s) );
100 
101  // Helper variables
102  double me2_over_s = md_*md_ / s;
103  double g2_squared_over_three = g2_*g2_ / 3.;
104 
105  // Total cross section in natural units (MeV^(-2))
106  double xs = (4. / marley_utils::pi) * std::pow(marley_utils::GF * Ec_cm, 2)
107  * (std::pow(g1_, 2) + (g2_squared_over_three - g1_*g2_)*me2_over_s
108  + g2_squared_over_three*(1. + std::pow(me2_over_s, 2)));
109 
110  // Multiply the single-electron cross section by the number of electrons
111  // present in this atom
113  xs *= atom_.Z();
114  return xs;
115 }

117 double marley::ElectronReaction::diff_xs(int pdg_a, double KEa,
118  double cos_theta_c_cm) const
119 {
120  // If we're asked for the wrong projectile, then just return zero
121  if ( pdg_a != pdg_a_ ) return 0.;
122 
123  // If we're below threshold, then just return zero
124  if ( KEa < KEa_threshold_ ) return 0.;
125 
126  // Mandelstam s (square of the total center of momentum frame energy)
127  double s = std::pow(ma_ + mb_, 2) + 2.*mb_*KEa;
128 
129  // CM frame ejectile total energy
130  double Ec_cm = (s + mc_*mc_ - md_*md_) / ( 2. * std::sqrt(s) );
131 
132  // Helper variables
133  double me2_over_s = md_*md_ / s;
134  double overall_factor = (2. / marley_utils::pi)
135  * std::pow(marley_utils::GF * Ec_cm, 2);
136  double terms = std::pow(g1_, 2) + g1_*g2_*me2_over_s*(cos_theta_c_cm - 1.)
137  + std::pow(g2_ * (1. + marley_utils::ONE_HALF*(1. - me2_over_s)
138  * (cos_theta_c_cm - 1.)), 2);
139 
140  // Compute and return the cross section
141  double diff_xsec = overall_factor * terms;
142 
143  // Multiply the single-electron cross section by the number of electrons
144  // present in this atom
146  diff_xsec *= atom_.Z();
147 
148  return diff_xsec;
149 }

151 // Creates an event object by sampling the appropriate quantities and
152 // performing kinematic calculations
153 marley::Event marley::ElectronReaction::create_event(int pdg_a, double KEa,
154  marley::Generator& gen) const
155 {
156  // If the projectile's PDG code doesn't match that stored in this object,
157  // complain and refuse to create an event.
158  if ( pdg_a != pdg_a_ ) throw marley::Error("Could"
159  " not create this event. The requested projectile PDG code "
160  + std::to_string(pdg_a) + " does not match the value "
161  + std::to_string(pdg_a_) + " stored in the Reaction object.");
162 
163  // Also complain if we're below threshold
164  if ( KEa < KEa_threshold_ ) throw marley::Error("Could not create"
165  " this event. The kinetic energy (" + std::to_string(KEa) + " MeV)"
166  " of the projectile is below the reaction threshold of "
167  + std::to_string(KEa_threshold_) + " MeV.");
168 
169  double s, Ec_cm, pc_cm, Ed_cm;
170  two_two_scatter(KEa, s, Ec_cm, pc_cm, Ed_cm);
171 
172  // Compute the maximum differential cross section to use for rejection
173  // sampling. To do this, we analytically solve for the value of
174  // cos_theta_c_cm (labeled cth below) for which the derivative of the
175  // differential cross section vanishes. This is an extremum of the function
176  // and might correspond to the maximum. If cth is within the allowed angular
177  // range, then it is considered alongside the two endpoints (COS_MIN and
178  // COS_MAX), and the largest of the three values (or just the endpoint values
179  // if cth is outside the allowed range) is chosen as the maximum.
180  double me2_over_s = md_*md_ / s;
181  double B = marley_utils::ONE_HALF * std::pow(g2_*(1. - me2_over_s), 2);
182  double A = g1_*g2_*me2_over_s + g2_*g2_*(1. - me2_over_s) - B;
183  double cth = -A / B;
184 
185  // Set the differential cross section to a huge negative value
186  // at cth. This value will be compared to those at the angular
187  // endpoints if cth does not lie in the allowed range. Otherwise,
188  // the correct value will replace this one.
189  double dxs_at_cth = std::numeric_limits<double>::lowest();
190 
191  double max = 0.;
192  if ( cth >= COS_MIN && cth <= COS_MAX ) {
193  dxs_at_cth = this->diff_xs(pdg_a, KEa, cth);
194  }
195 
196  double dxs_at_min = this->diff_xs(pdg_a, KEa, COS_MIN);
197  double dxs_at_max = this->diff_xs(pdg_a, KEa, COS_MAX);
198 
199  // Find the maximum value of the differential cross section
200  max = std::max( { dxs_at_min, dxs_at_max, dxs_at_cth } );
201 
202  // Sample a CM frame scattering cosine for the ejectile.
203  double cos_theta_c_cm = gen.rejection_sample(
204  [this, pdg_a, KEa](double ctheta) -> double
205  { return this->diff_xs(pdg_a, KEa, ctheta); }, COS_MIN, COS_MAX, max);
206 
207  // Sample a CM frame azimuthal scattering angle (phi) uniformly on [0, 2*pi).
208  // We can do this because the differential cross section is independent of
209  // the azimuthal angle.
210  double phi_c_cm = gen.uniform_random_double(0., marley_utils::two_pi, false);
211 
212  // Create and return the completed event object
213  // Note: electrons have spin 1/2 and positive intrinsic parity
214  return make_event_object( KEa, pc_cm, cos_theta_c_cm, phi_c_cm, Ec_cm, Ed_cm,
215  0., 1, marley::Parity(true) );
216 }
