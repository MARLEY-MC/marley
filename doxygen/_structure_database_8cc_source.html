73 std::set<int> marley::StructureDatabase::find_all_nuclides(
74  const std::string& filename, DecayScheme::FileFormat format)
75 {
76  if (format != DecayScheme::FileFormat::talys)
77  throw marley::Error(std::string("StructureDatabase::")
78  + "find_all_nuclides() is not implemented for"
79  + " formats other than TALYS.");
80 
81  // First line in a TALYS level dataset has fortran
82  // format (2i4, 2i5, 56x, i4, a2)
83  // General regex for this line:
84  static const std::regex nuclide_line("[0-9 ]{18} {56}[0-9 ]{4}.{2}");
85 
86  // Open the TALYs level data file for parsing
87  std::ifstream file_in(filename);
88 
89  // If the file doesn't exist or some other error
90  // occurred, complain and give up.
91  if (!file_in.good()) throw marley::Error(std::string("Could not")
92  + " read from the TALYS data file " + filename);
93 
94  std::string line; // String to store the current line
95  // of the TALYS file during parsing
96 
97  std::istringstream iss; // String stream used to parse the line
98 
99  double Z; // atomic number
100  double A; // mass number
101 
102  // Particle Data Group codes for each nuclide in the file
103  std::set<int> PDGs;
104 
105  // Loop through the data file, recording all nuclide PDGs found
106  while (std::getline(file_in, line)) {
107  if (std::regex_match(line, nuclide_line)) {
108  // Load the new line into our istringstream object for parsing. Reset
109  // the stream so that we start parsing from the beginning of the string.
110  iss.str(line);
111  iss.clear();
112 
113  // The first two entries on a TALYS nuclide line are Z and A.
114  iss >> Z >> A;
115 
116  PDGs.insert(marley_utils::get_nucleus_pid(Z, A));
117  }
118  }
119 
120  file_in.close();
121 
122  return PDGs;
123 }

125 marley::DecayScheme* marley::StructureDatabase::get_decay_scheme(
126  const int particle_id)
127 {
128  // If we already have the DecayScheme object stored in the lookup table,
129  // then just retrieve it
130  auto iter = decay_scheme_table_.find( particle_id );
131  if ( iter != decay_scheme_table_.end() ) return iter->second.get();
132  // If not, see if a data file for it is listed in the index
133  else {
134 
135  marley::TargetAtom ta_requested( particle_id );
136  MARLEY_LOG_DEBUG() << "Looking up structure data for " << ta_requested;
137 
138  if ( !loaded_structure_index_ ) this->load_structure_index();
139  auto ds_file_iter = decay_scheme_filenames_.find( particle_id );
140 
141  // If not, then just give up and return a null pointer
142  if ( ds_file_iter == decay_scheme_filenames_.end() ) return nullptr;
143 
144  // If a file is available, load all of the decay schemes present in it
145  // and add them to the lookup table. If we find the one we're looking
146  // for, return a pointer to it. Otherwise, print a warning, give up,
147  // and return a null pointer.
148  std::string ds_file_name = ds_file_iter->second;
149  auto& fm = marley::FileManager::Instance();
150  std::string full_ds_file_name = fm.find_file( ds_file_name );
151  std::ifstream ds_data_file( full_ds_file_name );
152  bool found_it = false;
153  auto temp_ds = std::make_unique< marley::DecayScheme >();
154  int loaded_nuclide_count = 0;
155  while ( ds_data_file >> *temp_ds ) {
156  int ds_pdg = temp_ds->pdg();
157  if ( particle_id == ds_pdg ) found_it = true;
158  this->add_decay_scheme( ds_pdg, temp_ds );
159  marley::TargetAtom ta( ds_pdg );
160  MARLEY_LOG_DEBUG() << "Added decay scheme for " << ta << " from "
161  << full_ds_file_name;
162  ++loaded_nuclide_count;
163  temp_ds = std::make_unique< marley::DecayScheme >();
164  }
165  if ( !found_it ) {
166  MARLEY_LOG_WARNING() << "Failed to load nuclear structure"
167  << " data for " << ta_requested << " from the"
168  << " file " << ds_file_name;
169  // Make a nullptr entry in the lookup table to avoid duplicate attempts
170  // to load the missing data
171  decay_scheme_table_[ particle_id ] = nullptr;
172  }
173  if ( loaded_nuclide_count > 0 ) {
174  MARLEY_LOG_INFO() << "Loaded structure data for "
175  << loaded_nuclide_count << " nuclides from the file "
176  << full_ds_file_name;
177  }
178 
179  // Return the pointer to the newly-loaded DecayScheme (or nullptr
180  // if it could not be loaded) via recursion
181  return this->get_decay_scheme( particle_id );
182  }
183 }

330 void marley::StructureDatabase::initialize_jpi_table() {
331 
332  // Instantiate the file manager and use it to find
333  // the data file containing the ground-state spin-parities
334  // for many nuclei
335  const auto& fm = marley::FileManager::Instance();
336  std::string full_jpi_file_name
337  = fm.find_file( jpi_data_file_name_ );
338 
339  if ( full_jpi_file_name.empty() ) {
340  throw marley::Error( "Could not find the MARLEY nuclear ground-state"
341  " spin-parity data file " + jpi_data_file_name_ + ". Please ensure that"
342  " the folder containing it is on the MARLEY search path."
343  " If needed, the folder can be appended to the MARLEY_SEARCH_PATH"
344  " environment variable." );
345  }
346 
347  MARLEY_LOG_INFO() << "Loading ground-state nuclear spin-parities from "
348  << full_jpi_file_name;
349 
350  std::ifstream table_file( full_jpi_file_name );
351  int nuc_pdg, twoJ;
352  marley::Parity Pi;
353  while ( table_file >> nuc_pdg >> twoJ >> Pi ) {
354  MARLEY_LOG_DEBUG() << "Nucleus with PDG code " << nuc_pdg
355  << " has spin-parity " << static_cast<double>( twoJ ) / 2.
356  << Pi;
357  jpi_table_[ nuc_pdg ] = std::pair<int, marley::Parity>( twoJ, Pi );
358  }
359 
360  // Set the flag saying we've initialized the table of ground-state spin-parities.
361  // This will avoid duplicate attempts at initialization.
362  initialized_gs_spin_parity_table_ = true;
363 
364  // Also initialize the table of nuclear fragment properties now that we have
365  // the needed information
366  using namespace marley_utils;
367  constexpr std::array< int, 6 > FRAGMENTS_TO_CONSIDER =
368  { NEUTRON, PROTON, DEUTERON, TRITON, HELION, ALPHA };
369 
370  for ( int f_pdg : FRAGMENTS_TO_CONSIDER ) {
371  // Temporary storage
372  int f_twoJ;
373  marley::Parity f_Pi;
374 
375  // Look up the spin-parity of the nuclear fragment (assumed to be emitted in
376  // its ground state) in the data table
377  marley::StructureDatabase::get_gs_spin_parity( f_pdg, f_twoJ, f_Pi );
378 
379  // Create a new entry in the table of nuclear fragments that should be
380  // considered during unbound nuclear de-excitations
381  marley::StructureDatabase::fragment_table_.emplace(
382  std::make_pair( f_pdg, marley::Fragment(f_pdg, f_twoJ, f_Pi) ));
383  }

387 void marley::StructureDatabase::load_structure_index() {
388 
389  // Instantiate the file manager and use it to find
390  // the index to the decay scheme data files
391  const auto& fm = marley::FileManager::Instance();
392  std::string full_index_file_name = fm.find_file( structure_index_filename_ );
393 
394  if ( full_index_file_name.empty() ) {
395  throw marley::Error( "Could not find the MARLEY structure data index file "
396  + structure_index_filename_ + ". Please ensure that"
397  " the folder containing it is on the MARLEY search path."
398  " If needed, the folder can be appended to the MARLEY_SEARCH_PATH"
399  " environment variable." );
400  }
401 
402  MARLEY_LOG_INFO() << "Loaded structure data index from "
403  << full_index_file_name;
404 
405  std::ifstream index_file( full_index_file_name );
406  int nuc_pdg;
407  std::string data_file_name;
408  while ( index_file >> nuc_pdg >> data_file_name ) {
409  MARLEY_LOG_DEBUG() << "Nucleus with PDG code " << nuc_pdg
410  << " has a tabulated decay scheme in the file " << data_file_name;
411  decay_scheme_filenames_[ nuc_pdg ] = data_file_name;
412  }
413 
414  // Avoid duplicate loading of the structure index by setting the
415  // "already loaded" flag
416  loaded_structure_index_ = true;
