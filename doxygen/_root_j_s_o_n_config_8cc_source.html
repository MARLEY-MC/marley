25 bool marley::RootJSONConfig::process_extra_source_types(
26  const std::string& type, const marley::JSON& source_spec, int pdg_code,
27  std::unique_ptr<marley::NeutrinoSource>& source) const
28 {
29  if (type != "th1" && type != "tgraph") return false;
30 
31  std::string tfile = source_get("tfile", source_spec, type.c_str(),
32  nullptr);
33  std::string namecycle = source_get("namecycle", source_spec,
34  type.c_str(), nullptr);
35 
36  if (type == "th1") {
37  auto th1 = marley_root::get_root_object<TH1>(tfile, namecycle);
38  source = marley_root::make_root_neutrino_source(pdg_code, th1);
39  MARLEY_LOG_INFO() << "Created a TH1 "
40  << marley_utils::neutrino_pdg_to_string(pdg_code)
41  << " source with parameters";
42  MARLEY_LOG_INFO() << " Emin = " << source->get_Emin() << " MeV";
43  MARLEY_LOG_INFO() << " Emax = " << source->get_Emax() << " MeV";
44  return true;
45  }
46 
47  else if (type == "tgraph") {
48  auto tg = marley_root::get_root_object<TGraph>(tfile, namecycle);
49  source = marley_root::make_root_neutrino_source(pdg_code, tg);
50  MARLEY_LOG_INFO() << "Created a TGraph "
51  << marley_utils::neutrino_pdg_to_string(pdg_code)
52  << " source with parameters";
53  MARLEY_LOG_INFO() << " Emin = " << source->get_Emin() << " MeV";
54  MARLEY_LOG_INFO() << " Emax = " << source->get_Emax() << " MeV";
55  return true;
56  }
57 
58  // We shouldn't ever get here, but I'll include something just in case.
59  return false;
60 }
