33 void marley::RootOutputFile::open() {
34  bool file_exists = check_if_file_exists(name_);
35 
36  std::string tfile_open_mode("recreate");
37  if (mode_ == Mode::OVERWRITE && file_exists && !force_) {
38  bool overwrite = marley_utils::prompt_yes_no(
39  "Overwrite ROOT file " + name_);
40  if (!overwrite) {
41  MARLEY_LOG_INFO() << "Cancelling overwrite of ROOT file \""
42  << name_ << '\"';
43  tfile_open_mode = std::string("update");
44  mode_ = Mode::RESUME;
45  }
46  }
47  else if (mode_ == Mode::RESUME) {
48  if (!file_exists) throw marley::Error("Cannot resume run. Could"
49  " not open the ROOT file \"" + name_ + '\"');
50  else tfile_open_mode = std::string("update");
51  }
52  else if (mode_ == Mode::APPEND) throw marley::Error("Cannot use the"
53  " \"append\" mode with the ROOT output format");
54 
55  file_ = std::make_unique<TFile>(name_.c_str(),
56  tfile_open_mode.c_str());
57 
58  // Check if there was a problem opening the file (e.g., pre-existing
59  // file that has the wrong format, etc.). If so, complain and quit.
60  if (file_->IsZombie()) throw marley::Error("Invalid format or other"
61  " error encountered while opening the ROOT file \"" + name_ + '\"');
62 
63  if (mode_ == Mode::OVERWRITE) {
64  // Create a ROOT tree to store the events
65  tree_ = new TTree("MARLEY_event_tree",
66  "Neutrino events generated by MARLEY");
67 
68  // We create a branch to store the events here, but set the branch
69  // address to nullptr. This will be fixed later when write_event()
70  // is called.
71  tree_->Branch("event", "marley::Event", nullptr);
72  }
73 
74  else if (mode_ == Mode::RESUME) {
75  file_->GetObject("MARLEY_event_tree", tree_);
76  if (!tree_) throw marley::Error("Cannot resume run. Could not find"
77  " a valid MARLEY event tree in the ROOT file \"" + name_ + '\"');
78 
79  MARLEY_LOG_INFO() << "Continuing previous run from ROOT file \""
80  << name_ << "\"\nwhich contains " << tree_->GetEntries()
81  << " events.";
82 
83  // Get previous RNG seed from the ROOT file
84  std::string* seed = nullptr;
85  file_->GetObject("MARLEY_seed", seed);
86 
87  if (!seed) MARLEY_LOG_WARNING() << "Unable to read"
88  << " random number generator seed saved to the ROOT file from"
89  << " the previous run.";
90  else MARLEY_LOG_INFO() << "The previous run was initialized using"
91  << " the random number generator seed " << *seed;
92  }
93 
94  else throw marley::Error("Unrecognized file mode encountered in"
95  " RootOutputFile::open()");
96 }

98 void marley::RootOutputFile::load_marley_headers() {
99  // Current (24 July 2016) versions of ROOT 6 require runtime loading
100  // of headers for custom classes in order to use dictionaries
101  // correctly. If we're running ROOT 6+, do the loading here, and give
102  // the user guidance if there are any problems.
103  // TODO: see if you can use the -inlineInputHeader to include the
104  // headers in the libMARLEY_ROOT library and then load them at
105  // runtime from there. This isn't very well documented, so your early
106  // efforts at doing this didn't work out.
107  static bool already_loaded = false;
108  if (!already_loaded) {
109  already_loaded = true;
110  if (gROOT->GetVersionInt() >= 60000) {
111  MARLEY_LOG_INFO() << "ROOT 6 or greater detected. Loading class"
112  << " information\nfrom headers \"marley/Particle.hh\""
113  << " and \"marley/Event.hh\"\n";
114  TInterpreter::EErrorCode* ec = new TInterpreter::EErrorCode();
115  gInterpreter->ProcessLine("#include \"marley/Particle.hh\"", ec);
116  if (*ec != 0) throw marley::Error("Error loading MARLEY header"
117  " Particle.hh. For MARLEY headers stored in /path/to/include/"
118  "marley/, please add /path/to/include to your"
119  " ROOT_INCLUDE_PATH environment variable and try again.");
120  gInterpreter->ProcessLine("#include \"marley/Event.hh\"");
121  if (*ec != 0) throw marley::Error("Error loading MARLEY header"
122  " Event.hh. For MARLEY headers stored in"
123  " /path/to/include/marley/, please add /path/to/include to"
124  " your ROOT_INCLUDE_PATH environment variable and try again.");
125  }
126  }

152 void marley::RootOutputFile::close(const marley::JSON& json_config,
153  const marley::Generator& gen, const long dummy)
154 {
155  // Write the event tree to the ROOT file, replacing the previous
156  // version if one exists. Avoid data loss by not deleting the
157  // previous version until the new version is completely written to
158  // disk.
159  file_->cd();
160  tree_->Write(tree_->GetName(), TTree::kWriteDelete);
161 
162  // Save the current state of the generator to the ROOT file in case
163  // we want to resume a run later
164  write_generator_state(json_config, gen, dummy);
165 
166  file_->Close();
167 }

169 bool marley::RootOutputFile::resume(std::unique_ptr<marley::Generator>& gen,
170  long& num_previous_events)
171 {
172  if (mode_ != Mode::RESUME) {
173  throw marley::Error("Cannot call Output"
174  "File::resume() for an output mode other than"
175  " \"resume\"");
176  return false;
177  }
178 
179  std::string* conf = nullptr;
180  file_->GetObject("MARLEY_config", conf);
181  if (!conf) {
182  throw marley::Error("Cannot resume run. Failed to load"
183  " JSON configuration from the ROOT file \"" + name_ + '\"');
184  return false;
185  }
186 
187  marley::JSON config = marley::JSON::load(*conf);
188 
189  std::string* state = nullptr;
190  file_->GetObject("MARLEY_state", state);
191  if (!state) {
192  throw marley::Error("Cannot resume run. Failed to load"
193  " random number generator state string from the ROOT file \""
194  + name_ + '\"');
195  return false;
196  }
197 
198  gen = this->restore_generator( config );
199  gen->seed_using_state_string(*state);
200 
201  if (!tree_) throw marley::Error("Cannot resume run. Error"
202  " accessing the MARLEY event tree stored in the ROOT file \""
203  + name_ + '\"');
204 
205  num_previous_events = tree_->GetEntries();
206 
207  return true;
208 }
