25 std::complex<double> coulomb_H_plus(int l, double eta, double rho) {
26 
27  // Enable the MARLEY gsl error handler upon using this function
28  // for the first time
29  static bool error_handler_set = false;
30  if ( !error_handler_set ) {
31  gsl_set_error_handler( &marley_gsl_error_handler );
32  error_handler_set = true;
33  }
34 
35  // H+ = G + i F
36  gsl_sf_result F, Fp, G, Gp;
37  double exp_F, exp_G;
38  int code = gsl_sf_coulomb_wave_FG_e(eta, rho, static_cast<double>(l), 0,
39  &F, &Fp, &G, &Gp, &exp_F, &exp_G);
40  if (code == GSL_EOVRFLW) {
41  double Fl = F.val * std::exp(exp_F);
42  double Gl = G.val * std::exp(exp_G);
43  return std::complex<double>(Gl, Fl);
44  }
45 
46  return std::complex<double>(G.val, F.val);
47 }
