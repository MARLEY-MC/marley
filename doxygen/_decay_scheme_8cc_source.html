32 // Returns a pointer to the level owned by this decay scheme object
33 // that has the closest excitation energy to E_level (E_level
34 // has units of MeV).
35 marley::Level* marley::DecayScheme::get_pointer_to_closest_level(
36  double E_level)
37 {
38  // If this decay scheme doesn't own any levels, return nullptr immediately
39  size_t num_levels = levels_.size();
40  if (num_levels == 0) return nullptr;
41 
42  // Search for the level whose energy is closest to the given value of E_level
43  size_t e_index = level_lower_bound_index(E_level);
44 
45  if (e_index == num_levels) {
46  // The given energy is greater than every level energy in our decay scheme.
47  // We will therefore assume that the desired level is the highest level.
48  // Its index is given by one less than the number of elements in the sorted
49  // vector, so subtract one from our previous result.
50  --e_index;
51  }
52  else if (e_index > 0) {
53  // If the calculated index does not correspond to the first element, we
54  // still need to check which of the two levels found (one on each side) is
55  // really the closest. Do so and reassign the index if needed.
56  if (std::abs(E_level - levels_.at(e_index)->energy())
57  > std::abs(E_level - levels_.at(e_index - 1)->energy()))
58  {
59  --e_index;
60  }
61  }
62 
63  // Return a pointer to the selected level object
64  return levels_.at(e_index).get();
65 }

71 void marley::DecayScheme::do_cascade(marley::Level& initial_level,
72  marley::Event& event, marley::Generator& gen, int qIon)
73 {
74  MARLEY_LOG_DEBUG() << "Beginning gamma cascade at level with energy "
75  << initial_level.energy() << " MeV";
76 
77  bool cascade_finished = false;
78 
79  marley::Level* p_current_level = &initial_level;
80 
81  const marley::MassTable& mt = marley::MassTable::Instance();
82 
83  while (!cascade_finished) {
84 
85  // Randomly select a gamma to produce
86  const marley::Gamma* p_gamma = p_current_level->sample_gamma(gen);
87  if (!p_gamma) {
88  MARLEY_LOG_DEBUG() << " this level does not have any gammas";
89  cascade_finished = true;
90  }
91  else {
92  p_current_level = p_gamma->end_level();
93  if (!p_current_level) {
94  throw marley::Error(std::string("This")
95  + "gamma does not have an end level. Cannot continue cascade.");
96  }
97  MARLEY_LOG_DEBUG() << std::setprecision(15) << std::scientific
98  << " emitted gamma with energy "
99  << p_gamma->energy() << " MeV. New level has energy "
100  << p_current_level->energy() << " MeV.";
101  MARLEY_LOG_DEBUG() << "gamma energy = " << p_gamma->energy();
102 
103  // Get the excitation energy of the end level. This will be added to
104  // the ground state mass of the nucleus to determine its
105  // post-gamma-emission mass.
106  double Exf = p_current_level->energy();
107 
108  // Create new particle objects to represent the emitted gamma and
109  // recoiling nucleus
110  marley::Particle gamma(marley_utils::PHOTON, 0);
111  int pdg = marley_utils::get_nucleus_pid(Z_, A_);
112  marley::Particle nucleus(pdg, mt.get_atomic_mass(pdg)
113  + Exf - qIon*mt.get_particle_mass(marley_utils::ELECTRON), qIon);
114 
115  // Sample a direction assuming that the gammas are emitted
116  // isotropically in the nucleus's rest frame.
117  // sample from [-1,1]
118  double gamma_cos_theta = gen.uniform_random_double(-1, 1, true);
119  // sample from [0,2*pi)
120  double gamma_phi = gen.uniform_random_double(0, 2*marley_utils::pi,
121  false);
122 
123  marley::Particle& residue = event.residue();
124 
125  // Determine the final energies and momenta for the recoiling nucleus and
126  // emitted gamma ray. Store them in the final state particle objects.
127  marley_kinematics::two_body_decay(residue, gamma, nucleus,
128  gamma_cos_theta, gamma_phi);
129 
130  // Update the residue for this event to take into account changes from
131  // gamma ray emission
132  residue = nucleus;
133 
134  // Add the new gamma to the event
135  event.add_final_particle(gamma);
136  }
137  }
138 
139  MARLEY_LOG_DEBUG() << "Finished gamma cascade at level with energy "
140  << p_current_level->energy();
141 }

153 void marley::DecayScheme::parse_talys(const std::string& filename) {
154  // First line in a TALYS level dataset has fortran
155  // format (2i4, 2i5, 56x, i4, a2)
156  // General regex for this line:
157  // std::regex nuclide_line("[0-9 ]{18} {56}[0-9 ]{4}.{2}");
158  std::string nuc_id = marley_utils::nuc_id(Z_, A_);
159 
160  // Make the last character of the nuc_id lowercase to follow the TALYS
161  // convention
162  nuc_id.back() = tolower(nuc_id.back());
163 
164  const std::regex nuclide_line("[0-9 ]{18} {57}" + nuc_id);
165 
166  // Open the TALYs level data file for parsing
167  std::ifstream file_in(filename);
168 
169  // If the file doesn't exist or some other error
170  // occurred, complain and give up.
171  if (!file_in.good()) throw marley::Error(std::string("Could not")
172  + " read from the TALYS data file " + filename);
173 
174  std::string line; // String to store the current line
175  // of the TALYS file during parsing
176 
177  bool found_decay_scheme = false;
178 
179  while (std::getline(file_in, line)) {
180  if (std::regex_match(line, nuclide_line)) {
181  found_decay_scheme = true;
182  break;
183  }
184  }
185 
186  if (!found_decay_scheme) throw marley::Error(std::string("Gamma")
187  + "decay scheme data (adopted levels, gammas) for "
188  + marley_utils::nucid_to_symbol(nuc_id)
189  + " could not be found in the TALYS data file " + filename);
190 
191  MARLEY_LOG_DEBUG() << "Gamma decay scheme data for " + nuc_id
192  << " found. Using TALYS dataset";
193  MARLEY_LOG_DEBUG() << line;
194 
195  // Dummy integer and number of excited levels for this nuclide
196  int dummy, num_excited_levels;
197 
198  // Read in the number of excited levels from the first line of data
199  std::istringstream iss(line);
200  iss >> dummy >> dummy >> dummy >> num_excited_levels;
201 
202  for (int l_idx = 0; l_idx <= num_excited_levels; ++l_idx) {
203 
204  // Get the next line of the file. This will be a discrete level record
205  std::getline(file_in, line);
206 
207  // Load the new line into our istringstream object for parsing. Reset
208  // the stream so that we start parsing from the beginning of the string.
209  iss.str(line);
210  iss.clear();
211 
212  // Read in this level's index, energy, spin, parity, and
213  // number of gamma transitions
214  int level_num, pi, num_gammas;
215  double level_energy, spin;
216  iss >> level_num >> level_energy >> spin >> pi >> num_gammas;
217 
218  // Compute two times the spin so that we can represent half-integer
219  // nuclear level spins as integers
220  int twoJ = std::round(2 * spin);
221 
222  // Create a parity object to use when constructing the level
223  marley::Parity parity = marley::Parity(pi);
224 
225  // Construct a new level object and add it to the decay scheme. Get
226  // a pointer to the newly-added level
227  marley::Level& current_level = add_level(marley::Level(level_energy,
228  twoJ, parity));
229 
230  for (int g_idx = 0; g_idx < num_gammas; ++g_idx) {
231 
232  // Get the next line of the file. This will be a gamma record
233  std::getline(file_in, line);
234 
235  // Load the new line into our istringstream object for parsing. Reset
236  // the stream so that we start parsing from the beginning of the string.
237  iss.str(line);
238  iss.clear();
239 
240  // Read in the index of the final level and branching ratio
241  // for this gamma transition
242  int gamma_final_level_num;
243  double br;
244  iss >> gamma_final_level_num >> br;
245 
246  // Process this gamma if it has a nonvanishing branching ratio
247  if (br > 0.) {
248 
249  marley::Level* final_level = levels_.at(gamma_final_level_num).get();
250 
251  // Compute the gamma ray's energy in MeV by subtracting the energy
252  // of the final level from the energy of the initial level
253  double gamma_energy = level_energy - final_level->energy();
254 
255  // Create the new Gamma object for the current level
256  current_level.add_gamma(gamma_energy, br, final_level);
257  }
258  }
259  }
260 
261  file_in.close();
262 }

264 void marley::DecayScheme::print_report(std::ostream& ostr) const {
265  // Cycle through each of the levels owned by this decay scheme
266  // object in order of increasing energy
267  for(const auto& lev : levels_) {
268  int twoj = lev->twoJ();
269  std::string spin = std::to_string(twoj / 2);
270  // If 2*J is odd, then the level has half-integer spin
271  if (twoj % 2) spin += "/2";
272  marley::Parity parity = lev->parity();
273 
274  ostr << "Level at " << lev->energy() << " MeV has spin-parity "
275  << spin << parity << '\n';
276 
277  std::vector<marley::Gamma>& gammas = lev->gammas();
278 
279  // Cycle through each of the gammas owned by the current level
280  // (according to the ENSDF specification, these will already be
281  // sorted in order of increasing energy)
282  for(const auto& g : gammas) {
283  ostr << " has a gamma with energy " << g.energy() << " MeV";
284  ostr << " (transition to level at "
285  << g.end_level()->energy() << " MeV)" << '\n';
286  ostr << " and relative intensity " << g.relative_intensity() << '\n';
287  }
288  }

293  std::string nuc_id = marley_utils::nuc_id(Z_, A_);

306 
307  ostr << caption_beginning + "}}\\\\\n";
308 
309  ostr << marley_utils::latex_table_2;
310 
311  ostr << caption_beginning + " -- \\textit{continued}}} \\\\\n";
312 
313  ostr << marley_utils::latex_table_3;
314 
315  // Cycle through each of the levels owned by this decay scheme
316  // object in order of increasing energy
317  for(const auto& lev : levels_) {
318 
319  std::string sp = lev->spin_parity_string();
320 
321  ostr << lev->energy() << " & " << sp << " & ";
322 
323  const auto& gammas = lev->gammas();
324 
325  // If there aren't any gammas for this level, finish writing
326  // the current row of the table. Add extra space between this
327  // level and the next one.
328  if (gammas.empty()) {
329  ostr << " & &";
330  // If this is the last row of the table, don't add extra space.
331  if (lev == levels_.back()) ostr << '\n';
332  else ostr << " \\\\ \\addlinespace[\\ExtraRowSpace]\n";
333  }
334 
335  // Cycle through each of the gammas owned by the current level
336  for(const auto& g : gammas) {
337  // If this is not the first gamma, add empty columns
338  // for the level energy and spin-parity
339  if (&g != &gammas.front()) ostr << " & & ";
340  // Output information about the current gamma
341  ostr << g.energy() << " & " << g.relative_intensity()
342  << " & " << g.end_level()->energy();
343  // Add vertical space after the final gamma row. Also prevent page breaks
344  // in the middle of a list of gammas by outputting a star at the end of
345  // each row except the final gamma row.
346  if (&g == &gammas.back()) {
347  // Don't add the extra row space for the very last row in the table
348  if (lev == levels_.back()) ostr << '\n';
349  else ostr << " \\\\ \\addlinespace[\\ExtraRowSpace]" << '\n';
350  }
351  else ostr << " \\\\*" << '\n';
352  }
353  }
354  ostr << marley_utils::latex_table_4 << '\n';
355 }

397  if (cit != levels_.cend()) level_f_idx = std::distance(levels_.cbegin(),
398  cit);
399  out << " " << level_f_idx << '\n';
400  }

404 void marley::DecayScheme::read_from_stream(std::istream& in) {
405 
406  levels_.clear();
407 
408  int num_levels;
409  in >> Z_ >> A_ >> num_levels;
410 
411  // If we had trouble parsing the decay scheme header, then
412  // just return the stream without doing anything else.
413  if ( !in ) return;
414 
415  double energy, ri;
416  int two_j, num_gammas, level_f_idx;
417  marley::Parity pi;
418 
419  for ( int i = 0; i < num_levels; ++i ) {
420  in >> energy >> two_j >> pi >> num_gammas;
421  marley::Level& l = add_level( marley::Level(energy, two_j, pi) );
422  for ( int j = 0; j < num_gammas; ++j ) {
423  in >> energy >> ri >> level_f_idx;
424  l.add_gamma( energy, ri, levels_.at(level_f_idx).get() );
425  }
426  }
