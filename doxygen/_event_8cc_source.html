243 void marley::Event::read(std::istream& in) {
244  this->clear();
245 
246  int num_initial;
247  int num_final;
248 
249  in >> num_initial >> num_final >> Ex_ >> twoJ_ >> parity_;
250 
251  // If reading the event header line failed for some
252  // reason, just return without trying to do anything else.
253  if ( !in ) return;
254 
255  // If we have invalid numbers of particles in either the initial or final
256  // state (there need to be at least two in each and we can't fill memory when
257  // allocating space for them) then complain by throwing an error
258  if ( num_initial < 2 || static_cast<size_t>(num_initial)
259  > initial_particles_.max_size() )
260  {
261  throw marley::Error("Invalid number of initial state particles ("
262  + std::to_string(num_initial) + ") encountered in marley::Event::read()");
263  }
264  else if ( num_final < 2 || static_cast<size_t>(num_final)
265  > final_particles_.max_size() )
266  {
267  throw marley::Error("Invalid number of final state particles ("
268  + std::to_string(num_final) + ") encountered in marley::Event::read()");
269  }
270 
271  initial_particles_.resize( num_initial );
272  final_particles_.resize( num_final );
273 
274  for (int i = 0; i < num_initial; ++i) {
275  marley::Particle* p = new marley::Particle;
276  in >> *p;
277  if ( !in ) throw marley::Error("Parse error while reading initial"
278  " particle #" + std::to_string(i) + " from an ASCII-format event"
279  " record");
280  initial_particles_.at(i) = p;
281  }
282  for (int f = 0; f < num_final; ++f) {
283  marley::Particle* p = new marley::Particle;
284  in >> *p;
285  if ( !in ) throw marley::Error("Parse error while reading final"
286  " particle #" + std::to_string(f) + " from an ASCII-format event"
287  " record");
288  final_particles_.at(f) = p;
289  }
290 }

292 // Function that dumps a marley::Particle to an output stream in HEPEVT format.
293 // This is a private helper function for the publicly-accessible write_hepevt.
294 void marley::Event::dump_hepevt_particle(const marley::Particle& p,
295  std::ostream& os, int status, int jmohep1, int jmohep2) const
296 {
297  // Print the status code to begin the particle entry in the HEPEVT record
298  os << status << ' ';
299 
300  // TODO: improve this entry to give the user more control over the vertex
301  // location and to reflect the parent-daughter relationships between
302  // particles.
303  // Convert from MARLEY natural units (MeV) to GeV for the HEPEVT format
304  os << p.pdg_code() << ' ' << jmohep1 << ' ' << jmohep2 << " 0 0 "
305  << p.px() / GEV_TO_MEV << ' ' << p.py() / GEV_TO_MEV
306  << ' ' << p.pz() / GEV_TO_MEV << ' ' << p.total_energy() / GEV_TO_MEV
307  << ' ' << p.mass() / GEV_TO_MEV
308  // The spacetime origin is currently used as the initial position 4-vector
309  // for all particles
310  << " 0. 0. 0. 0." << '\n';
311 }

313 void marley::Event::write_hepevt(size_t event_num, double flux_avg_tot_xsec,
314  std::ostream& out) const
315 {
316  // Use an temporary ostringstream object so that we can ensure all
317  // floating-point values are output with full precision without disturbing
318  // the user's settings in the "out" stream.
319  std::ostringstream temp;
320  temp.precision( std::numeric_limits<double>::max_digits10 );
321  temp << std::scientific;
322 
323  // Create a dummy particle that encodes extra MARLEY-specific information in
324  // the HEPEVT format. Preserve MARLEY natural units for these quantities
325  // (MeV) by pre-multiplying by the conversion factor used in
326  // dump_hepevt_particle()
327  marley::Particle dummy_particle;
328  dummy_particle.set_total_energy( Ex_ * GEV_TO_MEV );
329  dummy_particle.set_mass( flux_avg_tot_xsec * GEV_TO_MEV );
330 
331  // Add one to the total particle count so that our dummy particle will be
332  // included correctly
333  size_t num_particles = 1 + initial_particles_.size()
334  + final_particles_.size();
335 
336  // Write the HEPEVT header line to the event record
337  temp << event_num << ' ' << num_particles << '\n';
338 
339  // Write the initial particles to the event record
340  for (const auto i : initial_particles_) dump_hepevt_particle(*i, temp,
341  HEPEVT_INITIAL_STATE_STATUS_CODE);
342 
343  // Write our dummy particle to the event record
344  // We use the jmohep1 slot to record the twoJ_ data member and the jmohep2
345  // slot to record the parity_ data member (both as integers)
346  dump_hepevt_particle( dummy_particle, temp, HEPEVT_MARLEY_INFO_STATUS_CODE,
347  twoJ_, static_cast<int>(parity_) );
348 
349  // Write the final particles to the event record
350  for (const auto f : final_particles_) dump_hepevt_particle(*f, temp,
351  HEPEVT_FINAL_STATE_STATUS_CODE);
352 
353  // Output the finished HEPEVT format event to the "out" stream
354  out << temp.str();
355 }

375 // Reconstructs a marley::Event object from an input HEPEVT-format event record
376 // in an input stream. Returns a boolean that reflects whether the input stream
377 // state was still good at the end of the attempt to read the HEPEVT record.
378 // This can be used as a while loop condition when reading in multiple HEPEVT
379 // records.
380 bool marley::Event::read_hepevt(std::istream& in, double* flux_avg_tot_xsec)
381 {
382  // If flux_avg_tot_xsec is not null, then clear any previous value there
383  // before continuing. It will be loaded with a new value below if we
384  // find a dummy particle encoding extra MARLEY-specific information in
385  // the input HEPEVT record.
386  if ( flux_avg_tot_xsec ) *flux_avg_tot_xsec = 0.;
387 
388  // Remove any pre-existing contents in this event
389  this->clear();
390 
391  // Check that the event contains exactly two initial-state particles
392  size_t initial_state_particles = 0u;
393  // Check that one of the initial-state particles is an atom (ion)
394  size_t initial_state_ions = 0u;
395  // Check that the event contains exactly one final-state lepton
396  size_t final_state_leptons = 0u;
397  // Check that the event contains at least one final-state ion
398  size_t final_state_ions = 0u;
399 
400  // Initial indices of the final lepton and largest final ion (as judged
401  // by the mass number A) in the final_particles_
402  // vector. These will be used below to move them into the correct order.
403  size_t final_lepton_idx = 0u;
404  size_t largest_final_ion_idx = 0u;
405  int largest_A = 0;
406 
407  int event_num; // HEPEVT event number (ignored)
408  int num_particles; // Total number of particles stored in the event
409  in >> event_num >> num_particles;
410 
411  // If reading the first line failed for some
412  // reason, just return without trying to do anything else.
413  if ( !in ) return false;
414 
415  // If the number of particles in the event is negative or extremely
416  // huge (so that it would overload memory) then throw a marley::Error
417  if ( num_particles < 0 || static_cast<size_t>(num_particles)
418  > initial_particles_.max_size() )
419  {
420  throw marley::Error("Invalid number of particles ("
421  + std::to_string(num_particles) + ") encountered in marley::"
422  "Event::read_hepevt()");
423  }
424 
425  // Dummy variables used to skip HEPEVT fields that MARLEY doesn't
426  // care about (e.g., the particle production vertex 4-position)
427  int dummy_int;
428  double dummy_double;
429 
430  // Fields that we do care about
431  double Etot, px, py, pz, M;
432  int status_code, pdg, jmohep1, jmohep2;
433  for ( int p = 0; p < num_particles; ++p ) {
434 
435  in >> status_code >> pdg >> jmohep1 >> jmohep2;
436 
437  // Skip the JDAHEP1 and JDAHEP2 fields (daughter indices)
438  for ( int j = 0; j < 2; ++j ) in >> dummy_int;
439 
440  // Read in the particle 4-momentum and mass.
441  in >> px >> py >> pz >> Etot >> M;
442 
443  // Skip the VHEP1 through VHEP4 fields (production vertex 4-position)
444  for ( int j = 0; j < 4; ++j ) in >> dummy_double;
445 
446  if ( !in ) throw marley::Error("Parse error while reading particle #"
447  + std::to_string(p) + " from a HEPEVT-format event record");
448 
449  // If the particle has this status code, it is a dummy particle that
450  // contains MARLEY-specific information
451  if ( status_code == HEPEVT_MARLEY_INFO_STATUS_CODE ) {
452  // The dummy particle total energy stores the nuclear excitation energy.
453  // This value can be reconstructed (with some effort) from the
454  // particle four-vectors, but just storing it is a lot easier.
455  Ex_ = Etot;
456  // The dummy particle mass stores the flux-averaged total cross section.
457  // If the user passed something other than a nullptr as the
458  // flux_avg_tot_xsec argument to this function, then load it
459  // with this value, which is not normally stored in the marley::Event
460  // itself.
461  if ( flux_avg_tot_xsec ) *flux_avg_tot_xsec = M;
462  // The JMOHEP1 field contains two times the residue spin immediately
463  // following the initial two-two scattering reaction
464  twoJ_ = jmohep1;
465  // The JMOHEP2 field contains an integer representation of the residue
466  // parity immediately following the initial two-two scattering reaction
467  parity_ = jmohep2;
468  }
469 
470  // If the particle has a status code other than the two used by
471  // MARLEY, then don't bother to record the particle in the event object
472  if ( status_code != HEPEVT_INITIAL_STATE_STATUS_CODE
473  && status_code != HEPEVT_FINAL_STATE_STATUS_CODE ) continue;
474 
475  // Convert the particle 4-momentum components and mass from GeV (used in
476  // the HEPEVT format) to MARLEY natural units (MeV)
477  px *= GEV_TO_MEV;
478  py *= GEV_TO_MEV;
479  pz *= GEV_TO_MEV;
480  Etot *= GEV_TO_MEV;
481  M *= GEV_TO_MEV;
482 
483  // We already checked that the particle has a status code recognized
484  // by MARLEY above, so store it in the event object in the appropriate
485  // place.
486  if ( status_code == HEPEVT_INITIAL_STATE_STATUS_CODE ) {
487  initial_particles_.push_back(
488  new marley::Particle(pdg, Etot, px, py, pz, M) );
489  ++initial_state_particles;
490  if ( marley_utils::is_ion(pdg) ) ++initial_state_ions;
491  }
492  else {
493  // status_code == HEPEVT_FINAL_STATE_STATUS_CODE
494  final_particles_.push_back(
495  new marley::Particle(pdg, Etot, px, py, pz, M) );
496  if ( marley_utils::is_lepton(pdg) ) {
497  ++final_state_leptons;
498  final_lepton_idx = final_particles_.size() - 1;
499  }
500  if ( marley_utils::is_ion(pdg) ) {
501  ++final_state_ions;
502  // Update the index of the largest final ion
503  int A = marley_utils::get_particle_A( pdg ); // mass number
504  if ( A > largest_A ) {
505  largest_A = A;
506  largest_final_ion_idx = final_particles_.size() - 1;
507  }
508  }
509  }
510  }
511 
512  // Check that the event satisfies the criteria needed for a reasonable
513  // marley::Event
514  if ( initial_state_particles != 2u ) throw marley::Error(
515  std::to_string(initial_state_particles) + " initial-state particles"
516  " encountered while reading HEPEVT input");
517  else if ( initial_state_ions != 1u ) throw marley::Error("Multiple"
518  " initial-state ions encountered while reading HEPEVT input");
519  else if ( final_state_leptons != 1u ) throw marley::Error("Multiple"
520  " final-state leptons encountered while reading HEPEVT input");
521  else if ( final_state_ions < 1u ) throw marley::Error("Zero"
522  " final-state ions encountered while reading HEPEVT input");
523 
524  // Reorder (if needed) the initial particle list to ensure that the target
525  // (the initial atom) appears in the correct position. This is required
526  // because marley::Event objects rely on the ordering of the
527  // initial_particles_ array to allow easy retrieval of the projectile and
528  // target.
529  auto target_iter = initial_particles_.begin() + TARGET_INDEX;
530  if ( !marley_utils::is_ion( (*target_iter)->pdg_code() ) ) {
531  auto begin_ip = initial_particles_.begin();
532  std::iter_swap( begin_ip, begin_ip + 1 );
533  }
534 
535  // Play the same sort of game with the final particles. This will ensure that
536  // the ejectile and residue are stored in the correct places.
537  auto begin_fp = final_particles_.begin();
538  if ( final_lepton_idx != EJECTILE_INDEX ) {
539  std::iter_swap( begin_fp + EJECTILE_INDEX, begin_fp + final_lepton_idx );
540  }
541  if ( largest_final_ion_idx != RESIDUE_INDEX ) {
542  std::iter_swap( begin_fp + RESIDUE_INDEX, begin_fp + largest_final_ion_idx );
543  }
544 
545  // The marley::Particle constructor that we used above assumes that all ion
546  // PDG codes passed to it represent fully ionized particles. Here we will
547  // override this behavior and assume instead that the target atom has
548  // zero net charge in the initial state.
549  this->target().set_charge( 0 );
550 
551  // Determine the correct ionization state immediately following the prompt
552  // 2-->2 scattering reaction. Do this by assuming that the target atom is
553  // neutral (see above). This implies that the final ion charge must be equal
554  // and opposite to that of the final lepton.
555  int Qf_ion = -this->ejectile().charge();
556 
557  // Get the nuclear residue net electric charge by summing all final-state
558  // ions besides the residue itself. The difference between this sum and the
559  // pre-de-excitation ion charge is the charge that should be used.
560  int sum_Q_final_ions = 0;
561  for ( size_t f = 0u; f < final_particles_.size(); ++f ) {
562  if ( f == RESIDUE_INDEX ) continue; // skip the residue
563 
564  const auto& fp = final_particles_.at( f );
565  int pdg_f = fp->pdg_code();
566  if ( marley_utils::is_ion(pdg_f) ) sum_Q_final_ions += fp->charge();
567  }
568  this->residue().set_charge( Qf_ion - sum_Q_final_ions );
569 
570  // Everything was read in successfully. We're done!
571  return true;
572 }
