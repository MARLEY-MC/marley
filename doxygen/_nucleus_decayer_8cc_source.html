31  // In cases where no discrete level data are available, a continuum level
32  // density is used all the way down to the ground state. To avoid
33  // asymptotically approaching Ex = 0 in these cases, the de-excitation cascade
34  // will end once the excitation energy of the residual nucleus falls below
35  // this (small) value. Excitation energies below this value are considered
36  // "close enough" to the ground state for MARLEY not to worry about further
37  // de-excitations.
39  constexpr double CONTINUUM_GS_CUTOFF = 0.001; // MeV
40 
41  // The size of a tolerable discrepancy (in MeV) between the excitation energy
42  // stored in the event record and other (hopefully consistent) versions of it
43  constexpr double EX_TOLERANCE = 1e-5; // MeV
44 }

46 void marley::NucleusDecayer::process_event( marley::Event& event,
47  marley::Generator& gen )
48 {
49  // Get the residue excitation energy from the event. These values represent
50  // its state immediately following the initial two-two scattering reaction.
51  double Ex = event.Ex();
52  int twoJ = event.twoJ();
53  marley::Parity P = event.parity();
54 
55  // If the initial two-two scatter left the residue in its ground state,
56  // there's nothing for us to do. Just return without comment.
57  if ( Ex == 0. ) return;
58 
59  // The excitation energy should be nonnegative. Complain if it's not.
60  if ( Ex < 0. ) throw marley::Error("Negative excitation energy Ex = "
61  + std::to_string(Ex) + " MeV encountered in marley::NucleusDecayer::"
62  "deexcite_residue()");
63 
64  // To prevent accidental double application of the de-excitation cascade,
65  // check that the residue mass is consistent with the excitation energy
66  // stored in the event record (and thus was never decayed).
67  const auto& mt = marley::MassTable::Instance();
68  int initial_residue_pdg = event.residue().pdg_code();
69  int qIon = event.residue().charge();
70 
71  // Check that the residue PDG code makes sense. If it's not a nucleus,
72  // warn the user and refuse to do the cascade.
73  if ( !marley_utils::is_ion(initial_residue_pdg) ) {
74  MARLEY_LOG_WARNING() << "Unrecognized nuclear PDG code "
75  << initial_residue_pdg << " encountered in marley::NucleusDecayer::"
76  << "deexcite_residue(). The de-excitation cascade will be skipped";
77  return;
78  }
79 
80  double residue_mass = event.residue().mass();
81 
82  // Ground-state residue mass
83  double gs_residue_mass = mt.get_atomic_mass( initial_residue_pdg )
84  - qIon*mt.get_particle_mass( marley_utils::ELECTRON );
85 
86  double expected_residue_mass = gs_residue_mass + Ex;
87 
88  if ( std::abs(residue_mass - expected_residue_mass) > EX_TOLERANCE ) {
89 
90  if ( std::abs(residue_mass - gs_residue_mass) <= EX_TOLERANCE ) {
91  MARLEY_LOG_WARNING() << "Encountered ground-state nuclear remnant"
92  << " in marley::NucleusDecay::deexcite_residue(). The de-excitation"
93  << " cascade has already been applied.";
94  return;
95  }
96 
97  // If we get here, then the residue is not in its ground state but also
98  // not in the initial excited state given in the event record. Something
99  // went wrong with a partial application of a de-excitation cascade.
100  // Throw an error rather than trying to figure out how to do the right
101  // thing.
103  throw marley::Error("Partially de-excited nuclear remnant encountered"
104  " in marley::NucleusDecay::deexcite_residue().");
105  }
106 
107  // Decide whether we need to start the de-excitation cascade from a discrete
108  // nuclear level or from the continuum. Do this by comparing the excitation
109  // energy from the event record to the "unbound threshold" for the residue.
110  // If we're above the unbound threshold, do a continuum decay. Also start
111  // with a continuum decay if no discrete level data are available for the
112  // residue.
113  auto* ds = gen.get_structure_db().get_decay_scheme( initial_residue_pdg );
114  double unbound_threshold = mt.unbound_threshold( initial_residue_pdg );
115 
116  // If Reaction::set_level_ptrs() changes, you'll want to change this too.
117  // TODO: find a better way of keeping the two pieces of code in sync
118  bool continuum = ( Ex > unbound_threshold ) || ( !ds );
119 
120  // Keep track of whether the cascade was started from the continuum
121  // or not. If it was started from a discrete level, we'll double-check that
122  // discrete level's excitation energy below.
123  bool started_from_continuum = continuum;
124 
125  // Get a non-const reference to the nuclear residue. We'll use it to
126  // update the event record during each step of the Hauser-Feshbach
127  // cascade.
128  marley::Particle& residue = event.residue();
129 
130  if ( continuum ) {
131 
132  // Dummy particles used for temporary storage of binary decay products
133  // during the de-excitation cascade
134  marley::Particle first, second;
135 
136  // The selected level is unbound, so handle its de-excitation using
137  // the Hauser-Feshbach statistical model.
138  while ( continuum && Ex > CONTINUUM_GS_CUTOFF ) {
139 
140  auto& sdb = gen.get_structure_db();
141 
142  marley::HauserFeshbachDecay hfd( residue, Ex, twoJ, P, sdb );
143  MARLEY_LOG_DEBUG() << hfd;
144 
145  continuum = hfd.do_decay( Ex, twoJ, P, first, second, gen );
146 
147  MARLEY_LOG_DEBUG() << "Hauser-Feshbach decay to " << first.pdg_code()
148  << " and " << second.pdg_code();
149  MARLEY_LOG_DEBUG() << second.pdg_code() << " is at Ex = "
150  << Ex << " MeV.";
151 
152  residue = second;
153  event.add_final_particle( first );
154  }
155  }
156 
157  if ( !continuum ) {
158  // Either the selected initial level was bound (so it will only decay via
159  // gamma emission) or the Hauser-Feshbach decay process has now accessed a
160  // bound level in the residual nucleus. In either case, use gamma-ray decay
161  // scheme data to sample the de-excitation gammas and add them to this
162  // event's final particle list.
163  marley::DecayScheme* dec_scheme = gen.get_structure_db()
164  .get_decay_scheme( residue.pdg_code() );
165 
166  // Start the gamma cascade from this discrete level
167  marley::Level* lev = dec_scheme->get_pointer_to_closest_level( Ex );
168 
169  // If we get a null level pointer from the decay scheme, complain
170  if ( !lev ) throw marley::Error("Null nuclear level pointer encountered"
171  " in marley::NucleusDecayer::deexcite_residue()");
172 
173  // If we did not simulate any continuum decays before getting to this
174  // point, then double-check that the excitation energy from the event
175  // record and the initial level are consistent. If they're not, complain
176  // by throwing an error.
177  if ( !started_from_continuum ) {
178  double Ex_level = lev->energy();
179  if ( std::abs(Ex - Ex_level) > EX_TOLERANCE ) {
180  throw marley::Error("Excitation energy mismatch encountered in"
181  " marley::NucleusDecayer::deexcite_residue(). Event has Ex = "
182  + std::to_string(Ex) + " MeV while the initial discrete level has "
183  + std::to_string(Ex_level) + " MeV");
184  }
185  }
186 
187  dec_scheme->do_cascade( *lev, event, gen, residue.charge() );
188  }
