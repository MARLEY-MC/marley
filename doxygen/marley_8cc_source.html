71  // Show a number using one decimal digit without scientific notation.
72  // Used to print certain numbers in this way without affecting the settings
73  // currently in use for std::cout.
74  std::string format_number(double number) {
75  static std::stringstream temp_stream;
76  static bool configured = false;
77  if (!configured) {
78  temp_stream << std::fixed << std::setprecision(1);
79  }
80 
81  // Clear the std::stringstream of any of its previous contents
82  // This can be done more elegantly by move-assigning a default-constructed
83  // std::stringstream object like this:
84  // temp_stream = std::stringstream();
85  // However, g++ versions < 5 don't implement the move assignment
86  // operator for std::stringstream
87  // (see discussion at http://stackoverflow.com/q/27152263/4081973).
88  // The approach below is a reasonable workaround.
89  temp_stream.str("");
90  temp_stream.clear();
91 
92  temp_stream << number;
93 
94  return temp_stream.str();

122  // Use this instead of std::put_time to allow this executable to be built
123  // with g++ 4.9 (issue fixed in 5.0). See discussion here:
124  // http://stackoverflow.com/a/14137287
125  std::string put_time(std::tm* time, const char* format)
126  {
127  constexpr size_t TIME_STR_SIZE = 100;
128  std::string time_str(TIME_STR_SIZE, ' ');
129  // Pass a pointer to the first character in the time string as the char*
130  // argument of std::strftime(). This is well-defined behavior because the
131  // storage of std::string is guaranteed by the C++11 standard to be
132  // contiguous, and we have pre-allocated storage inside the string of the
133  // necessary size using the constructor above.
134  std::strftime(&time_str.front(), TIME_STR_SIZE, format, time);
135  marley_utils::trim_right_inplace(time_str);
136  return time_str;
137  }

139  // Formats the lines of the status display shown at the bottom of the
140  // screen when this executable is running
141  std::string makeStatusLines(long ev_count, long num_events, long num_old_events,
142  std::chrono::system_clock::time_point start_time_point,
143  const std::vector< std::unique_ptr<marley::OutputFile> >& output_files)
144  {
145  std::chrono::system_clock::time_point current_time_point
146  = std::chrono::system_clock::now();
147 
148  // Compute the average event generation rate (events / s) up to this point
149  auto elapsed_time = std::chrono::duration_cast<
150  marley_utils::seconds<double> >( current_time_point - start_time_point );
151  long events_since_start = ev_count - num_old_events;
152  double elapsed_seconds = elapsed_time.count();
153  double avg_event_rate = events_since_start / elapsed_seconds;
154 
155  std::ostringstream temp_oss;
156  // Print a status message showing the current number of events
157  double percent_complete = static_cast<double>( ev_count )
158  / num_events * 100.;
159  temp_oss << "\nEvent Count = " << ev_count << "/" << num_events
160  << " (" << format_number( percent_complete ) << "% complete, "
161  << format_number( avg_event_rate ) << " events / s)\033[K\n";
162 
163  // Print timing information
164  temp_oss << "Elapsed time: "
165  << marley_utils::elapsed_time_string(start_time_point,
166  current_time_point) << " (Estimated total run time: ";
167 
168  marley_utils::seconds<double> estimated_total_time =
169  (current_time_point - start_time_point)
170  * (static_cast<double>(num_events - num_old_events)
171  / (ev_count - num_old_events));
172 
173  temp_oss << marley_utils::duration_to_string
174  < marley_utils::seconds<double> >( estimated_total_time )
175  << ")\033[K\n";
176 
177  for (const auto& file : output_files) {
178  temp_oss << "Data written to " << file->name() << ' '
179  << marley_utils::num_bytes_to_string(file->bytes_written(), 2)
180  << "\033[K\n";
181  }
182 
183  std::time_t estimated_end_time = std::chrono::system_clock::to_time_t(
184  start_time_point + std::chrono::duration_cast
185  <std::chrono::system_clock::duration>( estimated_total_time ));
186 
187  temp_oss << "MARLEY is estimated to terminate on "
188  << put_time(std::localtime(&estimated_end_time), "%c %Z") << '\n';
189 
190  // Move up an extra line in the terminal for each output file because
191  // we're displaying the amount of data written to disk for each of them
192  // on a separate line.
193  for (size_t i = 0; i < output_files.size(); ++i) temp_oss << "\033[F";
194 
195  // Move up four lines
196  temp_oss << "\033[F\033[F\033[F\033[F";
197 
198  return temp_oss.str();
199  }

201  // Inserts an updated copy of the status lines after each
202  // newline is written to std::cout (e.g., by the marley::Logger
203  // class). Based on https://stackoverflow.com/a/22043916
204  class StatusInserter : public std::streambuf {
205  public:
206  StatusInserter(std::streambuf* dest, const long& ev_count,
207  const long& num_events, const long& num_old_events,
208  const std::chrono::system_clock::time_point& start_time_point,
209  const std::vector<std::unique_ptr<marley::OutputFile> >& output_files)
210  : std::streambuf(), myDest_( dest ), myIsAtStartOfLine_(true),
211  do_status_(true), ev_count_( ev_count ), num_events_( num_events ),
212  num_old_events_( num_old_events ), start_time_point_( start_time_point ),
213  output_files_( output_files ) {}
214 
215  inline void set_do_status(bool do_it) { do_status_ = do_it; }
216 
217  protected:
218  std::streambuf* myDest_;
219  bool myIsAtStartOfLine_;
220  bool do_status_;
221  const long& ev_count_;
222  const long& num_events_;
223  const long& num_old_events_;
224  const std::chrono::system_clock::time_point& start_time_point_;
225  const std::vector<std::unique_ptr<marley::OutputFile> >& output_files_;
226 
227  int overflow( int ch ) override {
228  int retval = 0;
229  if ( ch != traits_type::eof() ) {
230  if ( do_status_ && myIsAtStartOfLine_ ) {
231  std::string status = makeStatusLines(ev_count_,
232  num_events_, num_old_events_, start_time_point_, output_files_);
233  myDest_->sputn( status.data(), status.size() );
234  }
235  myIsAtStartOfLine_ = ch == '\n';
236  if ( myIsAtStartOfLine_ ) {
237  std::string erase( "\033[K" );
238  myDest_->sputn( erase.data(), erase.size() );
239  }
240  retval = myDest_->sputc( ch );
241  }
242  return retval;
243  }
244  };

251  std::streambuf* cerr_default_buf = std::cerr.rdbuf();

255  marley::Error::set_logging_status( false );

259  // Initialize the logger. Use a default log level of INFO. This will
260  // quickly be overwritten while parsing the JSON configuration file.
261  marley::Logger::Instance().add_stream(std::cout,
262  marley::Logger::LogLevel::INFO);
263 
264  // TODO: if you need command line parsing beyond the trivial stuff used
265  // here, consider using a header-only option parser library like cxxopts
266  // (https://github.com/jarro2783/cxxopts)
267  std::string config_file_name;
268 
269  // If the user has not supplied any command-line
270  // arguments, display the standard help message
271  // and exit
272  if (argc <= 1) {
273  print_help(argv[0]);
274  }
275  // The first command-line argument does not begin with a
276  // hyphen, so assume that it is the configuration file name.
277  else if (std::string(argv[1]).substr(0,1) != "-") {
278  config_file_name = argv[1];
279  }
280  // The first command-line argument begins with a hyphen,
281  // so treat it as an option and react accordingly.
282  // All of the current options cannot be combined, so
283  // just parse the first one and ignore the others.
284  else {
285  std::string option = argv[1];
286  if (option == "-h" || option == "--help") {
287  print_help(argv[0]);
288  }
289  else if (option == "-v" || option == "--version") {
290  print_version();
291  }
292  else if (option == "--marley") {
293  std::cout << marley_utils::marley_pic;
294  exit(0);
295  }
296  else {
297  std::cout << argv[0] << ": unrecognized "
298  << "command line option '" << option << "'\n";
299  print_help(argv[0]);
300  }
301  }
302 
303  // Parse the objects from the JSON-based configuration file
304  marley::JSON json = marley::JSON::load_file(config_file_name);
305 
306  // Process them to get the generator configuration. Enable ROOT support if
307  // it is available.
308  #ifdef USE_ROOT
309  marley::RootJSONConfig jc(json);
310  #else
311  marley::JSONConfig jc(json);
312  #endif
313 
314  // Get the time that the program was started
315  std::chrono::system_clock::time_point start_time_point
316  = std::chrono::system_clock::now();
317 
318  std::time_t start_time = std::chrono::system_clock::to_time_t(
319  start_time_point);
320 
321  std::cout << "\nMARLEY started on "
322  << put_time(std::localtime(&start_time), "%c %Z") << '\n';
323 
324  long num_old_events = 0;
325 
326  marley::JSON ex_set = json.get_object("executable_settings");
327 
328  // Desired number of events to be generated in this run. This
329  // will be read back from the ROOT file and overwritten
330  // if we're doing a continuation run.
331  long num_events = ex_set.get_long("events", 1e3);
332 
333  // Update the status messages at the bottom of the screen
334  // after this many events have been generated. The user may
335  // set a non-default value from the job configuration file.
336  int status_update_interval = DEFAULT_STATUS_UPDATE_INTERVAL;
337  if ( ex_set.has_key("status_update_interval") ) {
338  const auto& sui = ex_set.at( "status_update_interval" );
339 
340  bool ok;
341  int sui_value = sui.to_long( ok );
342 
343  // Check for settings that are not positive integers
344  if ( !ok || sui_value < 1 ) {
345  throw marley::Error( "Invalid value " + sui.dump_string()
346  + " given for the \"status_update_interval\" key in the"
347  " job configuration file" );
348  }
349  else status_update_interval = sui_value;
350  }
351 
352  std::vector<std::unique_ptr<marley::OutputFile> > output_files;
353 
354  if ( ex_set.has_key("output") ) {
355  marley::JSON output_set = ex_set.at("output");
356  if (!output_set.is_array()) throw marley::Error("The"
357  " \"output\" key in the executable settings must have a value that"
358  " is a JSON array.");
359  else for (const auto& el : output_set.array_range()) {
360  if (!el.has_key("file")) throw marley::Error("Missing file name"
361  " for an output file specification in the configuration file.");
362  else if (!el.has_key("format")) throw marley::Error("Missing format"
363  " for an output file specification in the configuration file.");
364 
365  std::string filename = el.at("file").to_string();
366  std::string format = el.at("format").to_string();
367 
368  std::string mode("overwrite"); // default mode is "overwrite"
369  if (el.has_key("mode")) mode = el.at("mode").to_string();
370 
371  bool force = false; // default behavior is to prompt before overwriting
372  if (el.has_key("force")) force = el.at("force").to_bool();
373 
374  // Set the indent level (if needed) for JSON format files
375  int indent = -1;
376  if (format == "json" && el.has_key("indent")) {
377  bool ok = false;
378  const marley::JSON& idt = el.at("indent");
379  indent = static_cast<int>(idt.to_long(ok));
380  if (indent < 0) ok = false;
381  if (!ok) {
382  throw marley::Error("Invalid indent value \""
383  + idt.dump_string() + "\" for JSON file \""
384  + filename + '\"');
385  indent = -1;
386  }
387  }
388 
389  #ifdef USE_ROOT
390  if (format == "root") output_files.push_back(
391  std::make_unique<marley::RootOutputFile>(filename, format, mode, force));
392  else output_files.push_back(std::make_unique<marley::TextOutputFile>(
393  filename, format, mode, force, indent));
394  #else
395  output_files.push_back(std::make_unique<marley::TextOutputFile>(filename,
396  format, mode, force, indent));
397  #endif
398  }
399  }
400  else {
401  // If the user didn't specify anything for the output key, then
402  // by default write to a single ASCII-format file.
403  output_files.push_back(std::make_unique<marley::TextOutputFile>("events.ascii",
404  "ascii", "overwrite", false));
405  }
406 
407  // This std::unique_ptr to a Generator object will be initialized below
408  // with the proper configuration.
409  std::unique_ptr<marley::Generator> gen;
410 
411  // Check if we need to resume from one of the files. If so, go ahead and
412  // do it. If more than one file is requested for the resume, complain.
413  bool need_to_resume = false;
414  for (auto& file : output_files) {
415  if (file->mode_is_resume()) {
416  if (need_to_resume) throw marley::Error("Only one file may be used"
417  " to resume a previous run.");
418  else {
419  need_to_resume = true;
420  bool resume_ok = file->resume(gen, num_old_events);
421  if (!resume_ok) throw marley::Error("Failed to resume previous run"
422  " from the file \"" + file->name() + '\"');
423  }
424  }
425  }
426 
427  // If we didn't resume a run from any of the files, use the current
428  // configuration
429  if (!need_to_resume) gen = std::make_unique<marley::Generator>(
430  jc.create_generator());
431 
432  // Use the signal handler defined above to deal with
433  // SIGINT signals (e.g., ctrl+c interruptions initiated
434  // by the user). This will allow us to terminate the
435  // loop gracefully, leaving a valid event tree file, etc.
436  std::signal(SIGINT, signal_handler);
437 
438  // Generate all of the requested events. End the loop early
439  // if the user interrupts execution (e.g., via ctrl+C)
440  auto event = std::make_unique<marley::Event>();
441  long ev_count = 1 + num_old_events;
442 
443  // Make std::cout use our "status inserter" std::streambuf
444  // object so that the status lines get automatically updated
445  // with every newline
446  StatusInserter my_status_inserter(cout_default_buf, ev_count,
447  num_events, num_old_events, start_time_point, output_files);
448  std::cout.rdbuf( &my_status_inserter );
449  std::cerr.rdbuf( &my_status_inserter );
450 
451  // Compute the flux-averaged total cross section for all
452  // enabled reactions
453  double avg_tot_xs = gen->flux_averaged_total_xs(); // MeV^(-2)
454 
455  // Write the flux-averaged total cross section to the output
456  // files (if needed)
457  for (auto& file : output_files) {
458  file->write_flux_avg_tot_xsec( avg_tot_xs );
459  }
460 
461  // Update the start time to just before we begin the event loop.
462  // This will help us get the best estimate for the remaining time
463  // that the program will run.
464  start_time_point = std::chrono::system_clock::now();
465  start_time = std::chrono::system_clock::to_time_t( start_time_point );
466 
467  for (; ev_count <= num_events && !interrupted; ++ev_count) {
468 
469  // Create an event using the generator object
470  *event = gen->create_event();
471 
472  for (const auto& file : output_files) {
473  file->write_event(event.get());
474  }
475 
476  // Print status messages about simulation progress after every
477  // status_update_interval events have been generated
478  if ( (ev_count - num_old_events) % status_update_interval == 1
479  || ev_count == num_events || status_update_interval == 1 )
480  {
481  // Temporarily disable the auto-printing of the status lines by
482  // the StatusInserter class. This will avoid duplication of the
483  // information.
484  my_status_inserter.set_do_status( false );
485 
486  // Print a status message showing the current number of events
487  std::cout << makeStatusLines(ev_count, num_events, num_old_events,
488  start_time_point, output_files);
489 
490  // Re-enable the auto-printing of the status lines now that we've
491  // printed them manually
492  my_status_inserter.set_do_status( true );
493  }
494  }
495 
496  // Restore the default std::streambuf to std::cout
497  std::cout.rdbuf( cout_default_buf );
498  std::cerr.rdbuf( cerr_default_buf );
499 
500  std::cout << "\033[E";
501 
502  for (const auto& file : output_files) {
503  file->close(json, *gen, ev_count - 1);
504  std::cout << "Data written to " << file->name() << ' '
505  << marley_utils::num_bytes_to_string(file->bytes_written())
506  << "\033[K\n";
507  }
508 
509  // Display the time that the program terminated
510  std::chrono::system_clock::time_point end_time_point
511  = std::chrono::system_clock::now();
512  std::time_t end_time = std::chrono::system_clock::to_time_t(
513  end_time_point);
514 
515  if (!interrupted) std::cout << "MARLEY terminated normally on ";
516  else std::cout << "MARLEY was interrupted by the user on ";
517  std::cout << put_time(std::localtime(&end_time), "%c %Z")
518  << "\033[K\033[E\033[K\n\033[K";
519 
520  return 0;
521  }
522 
523  catch ( const std::exception& error ) {
524  // Restore the old std::streambuf for std::cout
525  std::cout.rdbuf( cout_default_buf );
526  std::cerr.rdbuf( cerr_default_buf );
527 
528  // Flush the Logger, then rethrow the error for the system to handle.
529  // This will probably result in std::terminate() being called.
530  auto& log = marley::Logger::Instance();
531  log.flush();
532 
533  // We've flushed the logger, so we can now print the exception's error
534  // message without messing up the logger output.
535  MARLEY_LOG_ERROR() << error.what();
536  //throw error;
537  }
538 
539  // We shouldn't ever get here unless there was an error
540  return 1;
