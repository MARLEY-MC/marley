81 void marley::FragmentDiscreteExitChannel::compute_total_width() {
82 
83  int remnant_pdg = this->final_nucleus_pdg();
84  marley::OpticalModel& om = sdb_->get_optical_model( remnant_pdg );
85 
86  // Get information about the emitted fragment
87  const marley::Fragment& f = *sdb_->get_fragment( fragment_pdg_ );
88 
89  int two_s = f.get_two_s(); // two times the fragment spin
90  marley::Parity Pa = f.get_parity(); // intrinsic parity
91 
92  // Maximum possible excitation energy in the daughter nucleus after the
93  // fragment is emitted
94  double Exf_max = this->max_Exf();
95 
96  // Excitation energy of the final nuclear level
97  double Exf = final_level_.energy();
98 
99  // Initialize the total decay width to zero, just in case
100  width_ = 0.;
101 
102  if ( Exf >= Exf_max ) {
103  issue_Exf_warning( Exf, Exf_max );
104  return;
105  }
106 
107  // Total kinetic energy in the CM frame immediately after the binary decay
108  double total_KE_CM_frame = Exf_max - Exf;
109 
110  int twoJf = final_level_.twoJ();
111  marley::Parity Pf = final_level_.parity();
112 
113  for (int two_j = std::abs(twoJi_ - twoJf); two_j <= twoJi_ + twoJf;
114  two_j += 2)
115  {
116  int j_plus_s = (two_j + two_s) / 2;
117 
118  // For each new iteration, increment l and flip the overall final
119  // state parity
120  int l = std::abs(two_j - two_s) / 2;
121  bool l_is_odd = l % 2;
122  marley::Parity P_final_state = Pf * Pa * marley::Parity( !l_is_odd );
123  for ( ; l <= j_plus_s; ++l, !P_final_state )
124  {
125  // The current term in the sum only contributes to the total decay
126  // width if parity is conserved
127  if ( Pi_ == P_final_state ) {
128 
129  double Tlj = om.transmission_coefficient( total_KE_CM_frame,
130  fragment_pdg_, two_j, l, two_s );
131 
132  double partial_width = one_over_two_pi_rho_i_ * Tlj;
133 
134  width_ += partial_width;
135 
136  // TODO: cache term indexed by l, two_j
137  }
138  }
139  }
140 }

151 TrType marley::GammaExitChannel::get_transition_type( int mpol,
152  marley::Parity Pf ) const
153 {
154  // Electric transitions satisfy the parity relation Pi = (-1)^{\ell} * Pf,
155  // where Pi (Pf) are the initial (final) nuclear parities and \ell is the
156  // multipolarity of the transition. Magnetic transitions satisfy
157  // Pi = (-1)^{\ell + 1} * Pf. We choose the appropriate transition type
158  // here based on these rules.
159  bool mpol_is_odd = mpol % 2;
160  marley::Parity P_final_state = Pf * marley::Parity( !mpol_is_odd );
161  TrType type = ( Pi_ == P_final_state ) ? TrType::electric : TrType::magnetic;
162  return type;
163 }

165 void marley::GammaDiscreteExitChannel::compute_total_width() {
166 
167  // Retrieve the gamma strength function model used to compute transmission
168  // coefficients
169  marley::GammaStrengthFunctionModel& gsfm
170  = sdb_->get_gamma_strength_function_model( pdgi_ );
171 
172  // Get properties of the final nuclear level
173  double Exf = final_level_.energy();
174  int twoJf = final_level_.twoJ();
175  marley::Parity Pf = final_level_.parity();
176 
177  // Initialize the total width to zero, just in case.
178  width_ = 0.;
179 
180  if ( Exf >= Exi_ ) {
181  issue_Exf_warning( Exf, Exi_ );
182  return;
183  }
184 
185  // Compute the energy of the emitted gamma-ray
186  double E_gamma = this->gamma_energy( Exf );
187 
188  // Initialize the multipolarity for the start of the loop.
189  // There is no monopole radiation, so the minimum allowed value is one.
190  int start_mpol = std::max( 1, std::abs(twoJi_ - twoJf) / 2 );
191  int end_mpol = ( twoJi_ + twoJf ) / 2;
192 
193  // Sum contributions to the decay width from different multipolarities
194  for ( int mpol = start_mpol; mpol <= end_mpol; ++mpol ) {
195 
196  // Use the multipolarity and final-state nuclear parity to determine
197  // whether the current partial width represents an electric or magnetic
198  // transition
199  TrType type = this->get_transition_type( mpol, Pf );
200 
201  double TXl = gsfm.transmission_coefficient( type, mpol, E_gamma );
202 
203  double term = one_over_two_pi_rho_i_ * TXl;
204 
205  // TODO: add caching of terms by Xl pair
206 
207  width_ += term;
208  }
209 }

211 double marley::FragmentContinuumExitChannel::differential_width( double Exf,
212  bool store_jpi_widths ) const
213 {
214  if ( store_jpi_widths ) jpi_widths_table_.clear();
215 
216  int remnant_pdg = this->final_nucleus_pdg();
217  marley::OpticalModel& om = sdb_->get_optical_model( remnant_pdg );
218  marley::LevelDensityModel& ldm = sdb_->get_level_density_model( remnant_pdg );
219 
220  // Get the maximum accessible final excitation energy
221  double Exf_max = this->max_Exf();
222 
223  // Initialize the return value to zero
224  double diff_width = 0.;
225 
226  if ( Exf_max < E_c_min_ ) throw_continuum_bounds_error( E_c_min_, Exf_max );
227 
228  // Check that Exf lies within the continuum
229  if ( Exf < (E_c_min_ - TINY_OFFSET) || Exf > (Exf_max + TINY_OFFSET) ) {
230  // If it doesn't, complain and return zero
231  issue_Exf_continuum_warning( Exf, E_c_min_, Exf_max );
232  return 0.;
233  }
234 
235  double total_KE_CM_frame = Exf_max - Exf;
236 
237  // Get information about the emitted fragment
238  const marley::Fragment& f = *sdb_->get_fragment( fragment_pdg_ );
239 
240  int two_s = f.get_two_s(); // two times the fragment spin
241  marley::Parity Pa = f.get_parity(); // intrinsic parity
242 
243  // Final nuclear parity
244  marley::Parity Pf;
245  // The orbital parity starts as (-1)^0 = 1. Rather than applying parity
246  // conservation each time, just find the final state parity Pf for l = 0.
247  // Then we can safely flip Pf without further thought for each new l value in
248  // the loop.
249  if (Pi_ == Pa) Pf = 1;
250  else Pf = -1;
251  // For each new iteration, increment l and flip the final-state parity
252  for (int l = 0; l <= l_max_; ++l, !Pf) {
253  int two_l = 2*l;
254  for (int two_j = std::abs(two_l - two_s);
255  two_j <= two_l + two_s; two_j += 2)
256  {
257  for (int twoJf = std::abs(twoJi_ - two_j);
258  twoJf <= twoJi_ + two_j; twoJf += 2)
259  {
260  double Tlj = om.transmission_coefficient( total_KE_CM_frame,
261  fragment_pdg_, two_j, l, two_s );
262 
263  double rho_f = ldm.level_density( Exf, twoJf, Pf );
264 
265  double term = one_over_two_pi_rho_i_ * Tlj * rho_f;
266 
267  diff_width += term;
268 
269  if ( store_jpi_widths ) {
270  jpi_widths_table_.emplace_back( twoJf, Pf, term );
271  // TODO: include (l, two_j) in cached term
272  }
273  }
274  }
275  }
276  return diff_width;
277 }

306 double marley::GammaContinuumExitChannel::differential_width( double Exf,
307  bool store_jpi_widths ) const
308 {
309  if ( store_jpi_widths ) jpi_widths_table_.clear();
310 
311  auto& ldm = sdb_->get_level_density_model( pdgi_ );
312  auto& gsfm = sdb_->get_gamma_strength_function_model( pdgi_ );
313 
314  // Initialize the return value to zero
315  double diff_width = 0.;
316 
317  // Check that the continuum bounds make sense
318  if ( Exi_ < E_c_min_ ) throw_continuum_bounds_error( E_c_min_, Exi_ );
319 
320  // Check that Exf lies within the continuum
321  if ( Exf < (E_c_min_ - TINY_OFFSET) || Exf > (Exi_ + TINY_OFFSET) ) {
322  // If it doesn't, complain and return zero
323  issue_Exf_continuum_warning( Exf, E_c_min_, Exi_ );
324  return 0.;
325  }
326 
327  // Compute the energy of the emitted gamma-ray
328  double E_gamma = this->gamma_energy( Exf );
329 
330  // Array containing both possible parity values. It is used in
331  // the loop below.
332  constexpr std::array<marley::Parity, 2>
333  parities = { marley::Parity(true), marley::Parity(false) };
334 
335  // Sum over multipolarities. There is no monopole radiation, so
336  // the sum begins at mpol = 1.
337  for ( int mpol = 1; mpol <= l_max_; ++mpol ) {
338 
339  int two_mpol = 2 * mpol;
340 
341  for ( int twoJf = std::abs(twoJi_ - two_mpol); twoJf <= twoJi_ + two_mpol;
342  twoJf += 2 )
343  {
344  for ( const auto& Pf : parities ) {
345 
346  // Use the multipolarity and final-state nuclear parity to determine
347  // whether the current partial differential width represents an
348  // electric or magnetic transition
349  TrType type = this->get_transition_type( mpol, Pf );
350 
351  double Txl = gsfm.transmission_coefficient( type, mpol, E_gamma );
352  double rho_f = ldm.level_density(Exf, twoJf, Pf);
353 
354  double term = one_over_two_pi_rho_i_ * Txl * rho_f;
355 
356  if ( store_jpi_widths ) {
357  jpi_widths_table_.emplace_back( twoJf, Pf, term );
358  // TODO: include Xl in cached values
359  }
360 
361  diff_width += term;
362  }
363  }
364  }
365  return diff_width;
366 }

380 void marley::ExitChannel::prepare_products(
381  const marley::Particle& compound_nucleus, marley::Particle& emitted_particle,
382  marley::Particle& residual_nucleus, double Exf, marley::Generator& gen) const
383 {
384  const auto& mt = marley::MassTable::Instance();
385  int ep_pdg = this->emitted_particle_pdg();
386  double ep_mass = mt.get_particle_mass( ep_pdg );
387 
388  emitted_particle = marley::Particle( ep_pdg, ep_mass );
389 
390  // Proton number of the emitted particle
391  int ep_Z = marley_utils::get_particle_Z( ep_pdg );
392 
393  // Final ion charge after particle emission
394  int qf = qi_ - ep_Z;
395 
396  double me = mt.get_particle_mass( marley_utils::ELECTRON );
397  int remnant_pdg = this->final_nucleus_pdg();
398 
399  // Approximate the ground state mass of the ion formed when the fragment is
400  // emitted by subtracting qf electron masses from the atomic mass for the
401  // final nuclide.
402  double Mfgs_ion = mt.get_atomic_mass( remnant_pdg ) - qf*me;
403 
404  residual_nucleus = marley::Particle( remnant_pdg, Mfgs_ion + Exf, qf );
405 
406  // Now that the PDG codes, masses, and net charges of the binary decay
407  // products have been set, choose a direction for the emitted particle.
408  // TODO: Consider changing this to a more realistic model instead of
409  // isotropic emissions.
410  double cos_theta_emitted_particle
411  = gen.uniform_random_double( -1., 1., true );
412  double phi_emitted_particle
413  = gen.uniform_random_double( 0., marley_utils::two_pi, false );
414 
415  // Handle the kinematics calculations for this decay. Load the
416  // final-state particle objects with their full 4-momenta.
417  marley_kinematics::two_body_decay( compound_nucleus, emitted_particle,
418  residual_nucleus, cos_theta_emitted_particle, phi_emitted_particle );
419 }

421 double marley::ContinuumExitChannel::sample_Exf(marley::Generator& gen) const
422 {
423  // The maximum accessible excitation energy for this exit channel. It
424  // will be used when creating the ChebyshevInterpolatingFunction below
425  double Emax = this->E_c_max();
426 
427  // If we haven't built a CDF for sampling the final nuclear excitation
428  // energy yet, then build it before continuing
429  if ( !Exf_cdf_ ) {
430  // Build a polynomial approximant (at Chebyshev points) to the PDF for the
431  // final nuclear excitation energy
432  marley::ChebyshevInterpolatingFunction pdf_cheb( [this](double Exf)
433  -> double { return this->differential_width(Exf); }, E_c_min_, Emax,
434  marley::DEFAULT_N_CHEBYSHEV );
435 
436  // Store the cumulative density function for possible re-use
437  Exf_cdf_ = std::make_unique<marley::ChebyshevInterpolatingFunction>(
438  pdf_cheb.cdf() );
439  }
440 
441  // Sample a final nuclear excitation energy using the Chebyshev polynomial
442  // approximant to the CDF
443  double Exf = gen.inverse_transform_sample( *Exf_cdf_, E_c_min_, Emax );
444  return Exf;
445 }

465 void marley::ContinuumExitChannel::sample_spin_parity(double Exf, int& twoJ,
466  marley::Parity& Pi, marley::Generator& gen) const
467 {
468  // Clear any previous table entries of spin-parities and decay widths
469  jpi_widths_table_.clear();
470 
471  // Load table of partial differential widths via a call to
472  // differential_width()
473  double diff_width = this->differential_width( Exf, true );
474 
475  // Throw an error if all decays are impossible
476  if ( diff_width <= 0. ) throw marley::Error( "Cannot "
477  "continue Hauser-Feshbach decay. All partial differential decay widths"
478  "are zero." );
479 
480  // Sample a final spin and parity
481  const auto begin = marley::IteratorToMember<
482  std::vector<SpinParityWidth>::const_iterator,
483  const double>( jpi_widths_table_.cbegin(),
484  &SpinParityWidth::diff_width );
485 
486  const auto end = marley::IteratorToMember<
487  std::vector<SpinParityWidth>::const_iterator,
488  const double>( jpi_widths_table_.cend(),
489  &SpinParityWidth::diff_width );
490 
491  std::discrete_distribution<size_t> jpi_dist( begin, end );
492  size_t jpi_index = gen.sample_from_distribution( jpi_dist );
493 
494  // Store the results
495  const SpinParityWidth& Jpi = jpi_widths_table_.at( jpi_index );
496  twoJ = Jpi.twoJf;
497  Pi = Jpi.Pf;
498 }
