29 const marley::Gamma* marley::Level::sample_gamma(marley::Generator& gen)
30 {
31  if (gammas_.empty()) return nullptr;
32  else {
33  // Get the index of the gamma to return by randomly sampling from the
34  // discrete distribution gamma_dist using the standard marley_utils random
35  // number generator.
36  size_t g_index = gen.sample_from_distribution(gamma_dist_);
37  // Return a pointer to the corresponding gamma
38  return &(gammas_[g_index]);
39  }
40 }

81 void marley::Level::update_gamma_distribution() {
82  // Get iterators to the relative intensities of the gammas owned by this
83  // level.

85  auto ri_end = marley::Gamma::make_intensity_iterator(gammas_.end());

87  // Update the discrete distribution used to sample gammas
