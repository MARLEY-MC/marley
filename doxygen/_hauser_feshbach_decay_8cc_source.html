33 void marley::HauserFeshbachDecay::build_exit_channels(
34  marley::StructureDatabase& sdb)
35 {
36  // Remove any pre-existing ExitChannel objects, just in case
37  exit_channels_.clear();
38 
39  int pdgi = compound_nucleus_.pdg_code();
40  int Zi = marley_utils::get_particle_Z( pdgi );
41  int Ai = marley_utils::get_particle_A( pdgi );
42  int qi = compound_nucleus_.charge(); // Get net charge of initial ion
43 
44  // Get the initial nuclear level density (MeV^{-1}) in the vicinity of the
45  // initial nuclear level. This will be used to apply an overall normalization
46  // factor when computing exit channel decay widths. This isn't strictly
47  // needed for MC sampling, but it's helpful to work with physically
48  // meaningful units when possible.
49  marley::LevelDensityModel& ldm = sdb.get_level_density_model( Zi, Ai );
50  double rho_i = ldm.level_density( Exi_, twoJi_, Pi_ );
51 
52  total_width_ = 0.; // total compound nucleus decay width
53 
54  for ( const auto& pair : sdb.fragments() ) {
55 
56  const marley::Fragment& f = pair.second;
57 
58  // Get information about the current fragment
59  int two_s = f.get_two_s(); // spin
60  marley::Parity Pa = f.get_parity();
61  int fragment_pid = f.get_pid();
62  int Za = f.get_Z(); // atomic number
63  double Ma = f.get_mass(); // mass
64 
65  // Get information about the final-state nucleus
66  int Zf = Zi - f.get_Z(); // atomic number
67  int Af = Ai - f.get_A(); // mass number
68  int pdg_final = marley_utils::get_nucleus_pid(Zf, Af);
69 
70  // Approximate the ground state mass of the ion formed when the fragment f
71  // is emitted by adding (Za - qi) electron masses to the atomic mass for
72  // the final nucleus.
73  const auto& mt = marley::MassTable::Instance();
74  double Sa = mt.get_fragment_separation_energy( Zi, Ai, fragment_pid );
75 
76  // Get discrete level data (if any) and models for the final nucleus
77  marley::DecayScheme* ds = sdb.get_decay_scheme(pdg_final);
78 
79  // Determine the maximum excitation energy available after fragment
80  // emission in the final nucleus. This is simply the difference
81  // between the initial excitation energy and the fragment separation
82  // energy.
83  double Exf_max = Exi_ - Sa;
84 
85  // Check if emission of this fragment is energetically allowed. If we're
86  // exactly at threshold, still refuse to emit the fragment to avoid
87  // numerical problems.
88  if ( Exf_max <= 0. ) continue;
89 
90  // Let the continuum go down to 0 MeV unless there is a decay scheme object
91  // available for the final nuclide (we'll check this in a second).
92  double E_c_min = 0.;
93 
94  // If discrete level data are available for the final nuclide, get decay
95  // widths for each accessible level
96  if (ds) {
97 
98  // Get a vector of pointers to levels in the decay scheme. The levels are
99  // sorted in order of increasing excitation energy.
100  const auto& levels = ds->get_levels();
101 
102  // Use the maximum discrete level energy from the decay scheme object as
103  // the lower bound for the continuum
104  // TODO: consider whether this is the best approach
105  if (levels.size() > 0) E_c_min = levels.back()->energy();
106 
107  // Loop over the final discrete nuclear levels in order of increasing
108  // energy until the new level energy exceeds the maximum value. For each
109  // energetically allowed level, if a transition to it for a given
110  // fragment orbital angular momentum l and total angular momentum j
111  // conserves parity, then compute an optical model transmission
112  // coefficient and add it to the total.
113  for (const auto& level : levels) {
114  double Exf = level->energy();
115  if (Exf < Exf_max) {
116 
117  // Store information for this decay channel
118  auto ec = std::make_unique<marley::FragmentDiscreteExitChannel>(
119  pdgi, qi, Exi_, twoJi_, Pi_, rho_i, sdb, *level, f );
120 
121  total_width_ += ec->width();
122 
123  exit_channels_.push_back( std::move(ec) );
124  }
125  else break;
126  }
127  }
128 
129  // If transitions to the energy continuum are possible, include the
130  // continuum in the decay channels
131  if ( Exf_max > E_c_min ) {
132 
133  // Create an ExitChannel object to handle decays to the continuum
134  auto ec = std::make_unique<marley::FragmentContinuumExitChannel>(
135  pdgi, qi, Exi_, twoJi_, Pi_, rho_i, sdb, E_c_min, f );
136 
137  total_width_ += ec->width();
138 
139  exit_channels_.push_back( std::move(ec) );
140  }
141  }
142 
143  marley::DecayScheme* ds = sdb.get_decay_scheme( pdgi );
144 
145  // For gamma-ray emission, let the continuum go down to Ex = 0 MeV unless
146  // there is a decay scheme object available (we'll check this in a second).
147  double E_c_min = 0.;
148 
149  // If discrete level data is available for this nuclide, get gamma decay
150  // widths for each accessible level
151  if ( ds ) {
152 
153  // Loop over the final discrete nuclear levels in order of increasing
154  // energy until the new level energy exceeds the maximum value. For each
155  // energetically allowed level, compute a gamma ray transmission
156  // coefficient for it
157  const auto& levels = ds->get_levels();
158 
159  // Use the maximum discrete level energy from the decay scheme object as
160  // the lower bound for the continuum.
161  // TODO: consider whether this is the best approach
162  if ( levels.size() > 0 ) E_c_min = levels.back()->energy();
163 
164  for (const auto& level_f : levels) {
165  double Exf = level_f->energy();
166  if (Exf < Exi_) {
167  auto ec = std::make_unique<marley::GammaDiscreteExitChannel>( pdgi, qi,
168  Exi_, twoJi_, Pi_, rho_i, sdb, *level_f );
169 
170  total_width_ += ec->width();
171 
172  exit_channels_.push_back( std::move(ec) );
173  }
174  else break;
175  }
176  }
177 
178  // If gamma transitions to the energy continuum are possible, include them
179  // in the possible decay channels
180  if ( Exi_ > E_c_min ) {
181 
182  // Create an exit channel object to handle gamma-ray emission into the
183  // continuum
184  auto ec = std::make_unique<marley::GammaContinuumExitChannel>( pdgi, qi,
185  Exi_, twoJi_, Pi_, rho_i, sdb, E_c_min );
186 
187  total_width_ += ec->width();
188 
189  exit_channels_.push_back( std::move(ec) );
190  }
191 }

193 bool marley::HauserFeshbachDecay::do_decay(double& Exf, int& twoJf,
194  marley::Parity& Pf, marley::Particle& emitted_particle,
195  marley::Particle& residual_nucleus, marley::Generator& gen)
196 {
197  const auto& ec = this->sample_exit_channel( gen );
198 
199  ec->do_decay( Exf, twoJf, Pf, compound_nucleus_, emitted_particle,
200  residual_nucleus, gen );
201 
202  bool discrete_level = !ec->is_continuum();
203  return !discrete_level;
204 }

206 void marley::HauserFeshbachDecay::print(std::ostream& out) const {
207 
208  // Needed to print results in conventional units
209  constexpr double hbar = 6.58211951e-22; // MeV * s
210 
211  out << "Compound nucleus " << compound_nucleus_.pdg_code()
212  << " with Ex = " << Exi_ << ", spin = " << twoJi_ / 2;
213  if (twoJi_ % 2) out << ".5";
214  out << ", and parity = " << Pi_ << '\n';
215  out << "Total width = " << total_width_ << " MeV\n";
216  out << "Mean lifetime = " << hbar / total_width_ << " s\n";
217  for (const auto& ec : exit_channels_) {
218  double width = ec->width();
219  bool continuum = ec->is_continuum();
220  bool frag = ec->emits_fragment();
221  int pdg = ec->emitted_particle_pdg();
222  std::string symbol = marley_utils::particle_symbols.at( pdg );
223  out << " ";
224  if ( frag ) out << symbol;
225  else out << "gamma-ray";
226  if ( continuum ) out << " emission to the continuum width = ";
227  else {
228  auto* dec = dynamic_cast< marley::DiscreteExitChannel* >( ec.get() );
229  if ( !dec ) throw marley::Error( "Dynamic cast failed in marley::"
230  "HauserFeshbachDecay::print()" );
231  out << " emission to level at " << dec->get_final_level().energy()
232  << " MeV width = ";
233  }
234  out << width << " MeV\n";
235  }
236 }

238 const std::unique_ptr<marley::ExitChannel>&
239  marley::HauserFeshbachDecay::sample_exit_channel(
240  marley::Generator& gen) const
241 {
242  // Throw an error if all decays are impossible
243  if ( total_width_ <= 0. ) throw marley::Error("Cannot sample an exit channel"
244  " for a Hauser-Feshbach decay. All partial decay widths are zero.");
245 
246  // Sample an exit channel using a std::discrete distribution and a table of
247  // partial decay widths
248  const auto widths_begin
249  = marley::ExitChannel::make_width_iterator( exit_channels_.cbegin() );
250  const auto widths_end
251  = marley::ExitChannel::make_width_iterator( exit_channels_.cend() );
252 
253  std::discrete_distribution<size_t> exit_channel_dist(widths_begin,
254  widths_end);
255  size_t exit_channel_index = gen.sample_from_distribution( exit_channel_dist );
256 
257  const auto& ec = exit_channels_.at( exit_channel_index );
258  return ec;
259 }
