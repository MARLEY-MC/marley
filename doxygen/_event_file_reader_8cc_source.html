34 // Try to read an event from the file using each possible format. If we
35 // succeed, set the appropriate format code and return true. If all fail,
36 // return false.
37 bool marley::EventFileReader::deduce_file_format() {
38 
39  // Before we bother to check anything else, see if the file exists and is
40  // readable. Skip the MARLEY search path in this case (the event file should
41  // have been passed to the constructor with any needed path specification).
42  // Complain if the file cannot be read.
43  const auto& fm = marley::FileManager::Instance();
44  if ( fm.find_file(file_name_, "").empty() ) throw marley::Error("Could"
45  " not read from the file \"" + file_name_ + '\"');
46 
47  // Temporarily turn off logging of marley::Error messages for the
48  // try/catch blocks below. Otherwise, we'll end up with lots of noise
49  // in the Logger from this function.
50  bool log_marley_errors = marley::Error::logging_status();
51  marley::Error::set_logging_status( false );
52 
53  // Create a temporary event object to use for the following format checks
54  marley::Event temp_event;
55 
56 
57  // If the first character in the file is '{', then
58  // assume that the file is a JSON output file
59  in_.open( file_name_ );
60  char temp_char;
61  if ( in_ >> temp_char && temp_char == '{' ) {
62  format_ = marley::OutputFile::Format::JSON;
63 
64  // Reset the position of the input stream to before the initial '{'
65  in_.seekg(0);
66  return true;
67  }
68 
69  // If we can read in a marley::Event from the file via the input stream
70  // operator, then assume that the file is in ASCII format
71  // Reopen the file to reset the state flags and the stream cursor position
72  in_.clear();
73  in_.seekg(0);
74  try {
75  if ( in_ >> flux_avg_tot_xs_ >> temp_event ) {
76  format_ = marley::OutputFile::Format::ASCII;
77  in_.seekg(0);
78  return true;
79  }
80  }
81  // Ignore any marley::Error exceptions that were thrown. These can
82  // be expected if the file is not in ASCII format
83  catch (const marley::Error&) { }
84 
85  // Now try reading in a marley::Event assuming that the file is in HEPEVT
86  // format.
87  in_.clear();
88  in_.seekg(0);
89  try {
90  if ( temp_event.read_hepevt(in_, &flux_avg_tot_xs_) ) {
91  format_ = marley::OutputFile::Format::HEPEVT;
92  // Return to the start of the file so that we don't skip the first event
93  in_.seekg(0);
94  return true;
95  }
96  }
97  // Ignore any marley::Error exceptions that were thrown. These can
98  // be expected if the file is not in ASCII format
99  catch (const marley::Error&) { }
100 
101  // Now that we're past the blocks, restore the old marley::Error logging
102  // behavior
103  marley::Error::set_logging_status( log_marley_errors );
104 
105  // If everything else failed, then complain that events could not be read
106  return false;
107 }

109 void marley::EventFileReader::initialize() {
110  switch ( format_ ) {
111 
112  case marley::OutputFile::Format::ASCII:
113  in_ >> flux_avg_tot_xs_;
114  break;
115 
116  case marley::OutputFile::Format::HEPEVT:
117  // No further preparation is needed to read the HEPEVT format
118  break;
119 
120  case marley::OutputFile::Format::JSON: {
121 
122  // Turn off auto-logging of marley::Error objects so that
123  // we can add extra information via the try/catch block below
124  bool log_marley_errors = marley::Error::logging_status();
125  marley::Error::set_logging_status( false );
126 
127  try {
128  auto json = marley::JSON::load( in_ );
129 
130  flux_avg_tot_xs_ = json.at("gen_state").at("flux_avg_xsec")
131  .to_double();
132 
133  json_event_array_ = json.at("events");
134  json_event_array_wrapper_ = json_event_array_.array_range();
135  json_event_iter_ = json_event_array_wrapper_.begin();
136  // Move the iterator to "one before the beginning" so that
137  // we get the correct behavior in next_event() and
138  // operator bool()
139  --json_event_iter_;
140  }
141  catch (const std::exception& err) {
142  // Rethrow the error after adding commentary
143  MARLEY_LOG_ERROR() << "Parsing of a JSON-format output file"
144  << " failed with error \"" << err.what() << '\"';
145  throw err;
146  }
147 
148  // Restore the previous marley::Error logging behavior now that
149  // we're done with our mischief
150  marley::Error::set_logging_status( log_marley_errors );
151  break;
152  }
153 
154  default:
155  throw marley::Error("Unrecognized file format encountered in"
156  " marley::EventFileReader::initialize()");
157  }
158 }

160 bool marley::EventFileReader::next_event( marley::Event& ev )
161 {
162  this->ensure_initialized();
163  switch ( format_ ) {
164 
165  case marley::OutputFile::Format::ROOT:
166  MARLEY_LOG_WARNING() << "Cannot read from ROOT file."
167  << "Please use the marley::EventFileReader class and build"
168  << " MARLEY with ROOT support.";
169  return false;
170  break;
171 
172  case marley::OutputFile::Format::ASCII:
173  in_ >> ev;
174  if ( in_ ) return true;
175  break;
176 
177  case marley::OutputFile::Format::HEPEVT:
178  if ( ev.read_hepevt(in_, &flux_avg_tot_xs_) ) return true;
179  break;
180 
181  case marley::OutputFile::Format::JSON:
182  ++json_event_iter_;
183  if ( json_event_iter_ != json_event_array_wrapper_.end() ) {
184  ev.from_json( *json_event_iter_ );
185  return true;
186  }
187  break;
188 
189  default:
190  throw marley::Error("Unrecognized file format encountered in"
191  " marley::EventFileReader::next_event()");
192  }
193 
194  ev = marley::Event();
195  return false;
196 }

198 marley::EventFileReader::operator bool() const {
199 
200  switch ( format_ ) {
201 
202  case marley::OutputFile::Format::ROOT:
203  return false;
204  break;
205 
206  case marley::OutputFile::Format::ASCII:
207  case marley::OutputFile::Format::HEPEVT:
208  return static_cast<bool>( in_ );
209  break;
210 
211  case marley::OutputFile::Format::JSON:
212  return ( json_event_iter_ != json_event_array_wrapper_.end() );
213  break;
214 
215  default:
216  throw marley::Error("Unrecognized file format encountered in"
217  " marley::EventFileReader::operator bool()");
218  }
219 
220  return false;
221 }
